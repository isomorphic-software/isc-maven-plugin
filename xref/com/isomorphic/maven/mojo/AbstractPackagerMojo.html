72   * A base class meant to deal with prerequisites to install / deploy goals, which are basically
73   * to resolve the files in a given distribution to a collection of Maven artifacts suitable for
74   * installation or deployment to some Maven repository.

76   * The resulting artifacts are provided to this object's {@link #doExecute(Set)} method.

236 	/**
237 	 * Provides some initialization and validation steps around the collection and transformation of an Isomorphic SDK. 
238 	 * 
239 	 * @throws MojoExecutionException When any fatal error occurs.
240 	 * @throws MojoFailureException When any non-fatal error occurs. 
241 	 */
242 	public void execute() throws MojoExecutionException, MojoFailureException {
243 
244 		SimpleDateFormat dateFormat = new SimpleDateFormat("yyyy-MM-dd");
245 		dateFormat.setLenient(false);
246 		if (buildDate == null) {
247 			buildDate = dateFormat.format(new Date());
248 		}
249 		try {
250 			dateFormat.parse(buildDate);
251 		} catch (ParseException e) {
252 			throw new MojoExecutionException(String.format("buildDate '%s' must take the form yyyy-MM-dd.", buildDate));
253 		}
254 
255 		LOGGER.debug("buildDate set to '{}'", buildDate);
256 
257 		String buildNumberFormat = "\\d.*\\.\\d.*[d|p]";
258 		if (! buildNumber.matches(buildNumberFormat)) {
259 			throw new MojoExecutionException(String.format("buildNumber '%s' must take the form [major].[minor].[d|p].  e.g., 4.1d", buildNumber, buildNumberFormat));
260 		}
261 		
262 		File basedir = FileUtils.getFile(workdir, product.toString(), license.toString(), buildNumber, buildDate);
263 
264 		//add optional modules to the list of downloads
265 		List<License> licenses = new ArrayList<License>();
266 		licenses.add(license);
267 		if (license == POWER || license == ENTERPRISE) {
268 			if (includeAnalytics) {
269 				licenses.add(ANALYTICS_MODULE);
270 			}
271 			if (includeMessaging) {
272 				licenses.add(MESSAGING_MODULE);
273 			}
274 		}
275 
276 		//collect the maven artifacts and send them along to the abstract method
277 		Set<Module> artifacts = collect(licenses, basedir);
278 		
279 		File bookmarkable = new File(basedir.getParent(), "latest");
280 		LOGGER.info("Copying distribution to '{}'", bookmarkable.getAbsolutePath());
281 		try {
282 			FileUtils.forceMkdir(bookmarkable);
283 			FileUtils.cleanDirectory(bookmarkable);
284 			FileUtils.copyDirectory(basedir, bookmarkable, FileFilterUtils.notFileFilter(FileFilterUtils.nameFileFilter("zip")));
285 		} catch (IOException e) {
286 			throw new MojoFailureException("Unable to copy distribution contents", e);
287 		}
288 		
289 		doExecute(artifacts);
290 	
291 	}

293 	/**
294 	 * Download the specified distributions, if necessary, extract resources from them, and use the results to create Maven artifacts as appropriate:
295 	 * <p/>
296 	 * Try to install all of the main artifacts - e.g., those found in lib/*.jar and assembly/*.zip <br/>
297 	 * Try to match main artifacts to 'subartifacts' by name and attach them (with classifiers as necessary)
298 	 * 
299 	 * @param downloads The list of licenses top be included in the distribution.  Multiple licenses are only allowed to support the inclusion of optional modules.
300 	 * @param basedir The directory into which results should be written
301 	 * @return A collection of Maven artifacts resulting from the download and preparation of a supported Isomorphic SDK.
302 	 * @throws MojoExecutionException When any fatal error occurs.
303 	 */
304 	private Set<Module> collect(List<License> downloads, File basedir) throws MojoExecutionException {
305 		
306 		//allow execution to proceed without login credentials - it may be that they're not required
307 		Server server = settings.getServer(serverId);
308 		String username = null;
309 		String password = null;
310 		if (server != null) {
311 			username = server.getUsername();
312 			password = server.getPassword();
313 		} else {
314 			LOGGER.warn("No server configured with id '{}'.  Will be unable to authenticate.", serverId);
315 		}
316 		
317 		UsernamePasswordCredentials credentials = null; 
318 		if (username != null) {
319 			credentials = new UsernamePasswordCredentials(username, password);
320 		}		
321 
322 		File downloadTo = new File(basedir, "zip");
323 		downloadTo.mkdirs();
324 		
325 		Downloads downloadManager = new Downloads(credentials);
326 		downloadManager.setToFolder(downloadTo);
327 		downloadManager.setProxyConfiguration(settings.getActiveProxy());
328 		downloadManager.setOverwriteExistingFiles(overwrite);
329 
330 		File[] existing = downloadTo.listFiles();
331 		List<Distribution> distributions = new ArrayList<Distribution>();
332 		try {
333 			if (!skipDownload) {
334 				distributions.addAll(downloadManager.fetch(product, buildNumber, buildDate, downloads.toArray(new License[0])));
335 			} else if (existing != null) {
336 				LOGGER.info("Creating local distribution from '{}'", downloadTo.getAbsolutePath());
337 				Distribution distribution = Distribution.get(product, license, buildNumber, buildDate);
338 				distribution.getFiles().addAll(Arrays.asList(existing));
339 				distributions.add(distribution);
340 			}
341 			
342 			if (!skipExtraction) {
343 				LOGGER.info("Unpacking downloaded file/s to '{}'", basedir);
344 				for (Distribution distribution : distributions) {
345 					distribution.unpack(basedir);								
346 				}	
347 			}
348 			
349 			//it doesn't strictly read this way, but we're looking for lib/*.jar, pom/*.xml, assembly/*.zip
350 			//TODO it'd be better if this didn't have to know where the files were located after unpacking
351 			Collection<File> files = FileUtils.listFiles(basedir, 
352 				FileFilterUtils.or(FileFilterUtils.suffixFileFilter("jar"), FileFilterUtils.suffixFileFilter("xml"), FileFilterUtils.suffixFileFilter("zip")), 
353 				FileFilterUtils.or(FileFilterUtils.nameFileFilter("lib"), FileFilterUtils.nameFileFilter("pom"), FileFilterUtils.nameFileFilter("assembly"))
354 			);
355 
356 			if (files.isEmpty()) {
357 				throw new MojoExecutionException(String.format("There don't appear to be any files to work with at '%s'.  Check earlier log entries for clues.", basedir.getAbsolutePath()));
358 			}
359 			
360 			Set<Module> result = new TreeSet<Module>();
361 			for (File file : files) {
362 				try {
363 
364 					String base = FilenameUtils.getBaseName(file.getName().replaceAll("_", "-"));
365 
366 					//poms don't need anything else
367 					if ("xml".equals(FilenameUtils.getExtension(file.getName()))) {
368 						result.add(new Module(getModelFromFile(file)));
369 						continue;
370 					}
371 					
372 					//for each jar/zip, find the matching pom
373 					IOFileFilter filter = new WildcardFileFilter(base + ".pom");
374 					Collection<File> poms = FileUtils.listFiles(basedir, filter, TrueFileFilter.INSTANCE);
375 					if (poms.size() != 1) {
376 						LOGGER.warn("Expected to find exactly 1 POM matching artifact with name '{}', but found {}.  Skpping installation.", base, poms.size());
377 						continue;
378 					}
379 					
380 					Model model = getModelFromFile(poms.iterator().next());
381 					Module module = new Module(model, file);
382 					
383 					/*
384 					 * Find the right javadoc bundle, matched on prefix.  e.g., 
385 					 * 		smartgwt-eval -> smartgwt-javadoc
386 					 * 		isomorphic-core-rpc -> isomorphic-javadoc
387 					 * and add it to the main artifact with the javadoc classifier.  This seems appropriate as long as
388 					 * 		a) there is no per-jar javadoc
389 					 * 		b) naming conventions are adhered to (or can be corrected by plugin at extraction)
390 					 */
391 					int index = base.indexOf("-");
392 					String prefix = base.substring(0, index);
393 					
394 					Collection<File> doc = FileUtils.listFiles(new File(basedir, "doc"), 
395 						FileFilterUtils.prefixFileFilter(prefix), FileFilterUtils.nameFileFilter("lib"));
396 					
397 					if (doc.size() != 1) {
398 						LOGGER.debug("Found {} javadoc attachments with prefix '{}'.  Skipping attachment.", doc.size(), prefix);
399 					} else {
400 						module.attach(doc.iterator().next(), "javadoc");
401 					}
402 					
403 					result.add(module);
404 				} catch (ModelBuildingException e) {
405 					throw new MojoExecutionException("Error building model from POM", e);
406 				}
407 			}
408 			return result;
409 		} catch (IOException e) {
410 			throw new MojoExecutionException("Failure during assembly collection", e);
411 		}
412 	}

414 	/**
415 	 * Read the given POM so it can be used as the source of coordinates, etc. during artifact construction.  Note that
416 	 * if this object's {@link #snapshots} property is true, and we're working with a development build ({@link #buildNumber} 
417 	 * ends with 'd'), the POM is modified to remove the SNAPSHOT qualifier.
418 	 * 
419 	 * @param pom the POM file containing the artifact metadata
420 	 * @return A Maven model to be used at {@link com.isomorphic.maven.packaging.Module#Module(Model)} Module construction
421 	 * @throws ModelBuildingException if the Model cannot be built from the given POM
422 	 * @throws IOException if the Model cannot be built from the given POM
423 	 */
424 	private Model getModelFromFile(File pom) throws ModelBuildingException, IOException {
425 
426 		if (buildNumber.endsWith("d") && !snapshots) {
427 			LOGGER.info("Rewriting file to remove SNAPSHOT qualifier from development POM '{}'", pom.getName());
428 			String content = FileUtils.readFileToString(pom);
429 	        content = content.replaceAll("-SNAPSHOT", "");
430 	        FileUtils.write(pom, content);
431 		}
432 		
433 		ModelBuildingRequest request = new DefaultModelBuildingRequest();
434 		request.setPomFile(pom);
435 		
436 		ModelBuildingResult result = modelBuilder.build(request);
437 		return result.getEffectiveModel();
438 	}
