67  /**
68   * A base class meant to deal with prerequisites to install / deploy goals,
69   * which are basically to resolve the files in a given distribution to a
70   * collection of Maven artifacts suitable for installation or deployment to some
71   * Maven repository.
72   * <p>
73   * The resulting artifacts are provided to this object's {@link #doExecute(Set)}
74   * method.
75   */

256     /**
257      * Provides some initialization and validation steps around the collection
258      * and transformation of an Isomorphic SDK.
259      * 
260      * @throws MojoExecutionException
261      *             When any fatal error occurs.
262      * @throws MojoFailureException
263      *             When any non-fatal error occurs.
264      */
265     public void execute() throws MojoExecutionException, MojoFailureException {
266 
267         // allow execution to proceed without login credentials - it may be that
268         // they're not required
269         UsernamePasswordCredentials credentials = getCredentials(serverId);
270         if (credentials == null) {
271         	LOGGER.warn("No server configured with id '{}'.  Will be unable to authenticate.", serverId);
272         }
273         
274         String buildNumberFormat = "\\d.*\\.\\d.*[d|p]";
275         if (!buildNumber.matches(buildNumberFormat)) {
276             throw new MojoExecutionException(String.format(
277                 "buildNumber '%s' must take the form [major].[minor].[d|p].  e.g., 4.1d",
278                 buildNumber, buildNumberFormat));
279         }
280 
281         httpWorker = new HttpRequestManager(HOST, credentials, settings.getActiveProxy());
282         Downloads dl = new Downloads(httpWorker);
283         
284         try {
285 
286         	httpWorker.login();
287         	
288 	        SimpleDateFormat dateFormat = new SimpleDateFormat("yyyy-MM-dd");
289 	        dateFormat.setLenient(false);
290 	        if (buildDate == null) {
291 	            Distribution d = Distribution.get(product, license);
292 	            
293 	            LOGGER.info("No buildDate provided.  Contacting Isomorphic build server to "
294 	                + "look for the most recent distribution...");
295 	            
296 	            String link = dl.findCurrentBuild(d, buildNumber);
297 	            
298 	            if (link == null) {
299 	                throw new MojoExecutionException("No build found for the given distribution.");
300 	            }
301 	
302 	            LOGGER.debug("Extracting date from server response: '{}'", link);
303 	            
304 	            buildDate = StringUtils.substringAfterLast(link, "/");
305 	
306 	            LOGGER.info("buildDate set to '{}'", buildDate);
307 	            
308 	        }
309 
310             dateFormat.parse(buildDate);
311             
312 	        File basedir = FileUtils.getFile(workdir, product.toString(), license.toString(),
313 	            buildNumber, buildDate);
314 	
315 	        // add optional modules to the list of downloads
316 	        List<License> licenses = new ArrayList<License>();
317 	        licenses.add(license);
318 	        if (license == POWER || license == ENTERPRISE) {
319 	            if (includeAnalytics) {
320 	                licenses.add(ANALYTICS_MODULE);
321 	            }
322 	            if (includeMessaging) {
323 	                licenses.add(MESSAGING_MODULE);
324 	            }
325 	        }
326 	
327 	        // collect the maven artifacts and send them along to the abstract method
328 	        Set<Module> artifacts = collect(licenses, basedir);
329 
330 	        String[] executables = { "bat", "sh", "command" };
331 	        Collection<File> scripts = FileUtils.listFiles(basedir, executables, true);
332 	        
333 	        if (copyToLatestFolder) {
334 		        File bookmarkable = new File(basedir.getParent(), "latest");
335 		        LOGGER.info("Copying distribution to '{}'", bookmarkable.getAbsolutePath());
336 		        try {
337 		            FileUtils.forceMkdir(bookmarkable);
338 		            FileUtils.cleanDirectory(bookmarkable);
339 		            FileUtils.copyDirectory(basedir, bookmarkable,
340 		                FileFilterUtils.notFileFilter(FileFilterUtils.nameFileFilter("zip")));
341 		            
342 		            scripts.addAll(FileUtils.listFiles(bookmarkable, executables, true));
343 		            
344 		        } catch (IOException e) {
345 		            throw new MojoFailureException("Unable to copy distribution contents", e);
346 		        }	        	
347 	        }
348 	
349 	        for (File script : scripts) {
350 	            script.setExecutable(true);
351 	            LOGGER.debug("Enabled execute permissions on file '{}'", script.getAbsolutePath());
352 	        }
353         
354 	        doExecute(artifacts);
355 	        
356         } catch (ParseException e) {
357             throw new MojoExecutionException(String.format(
358                 "buildDate '%s' must take the form yyyy-MM-dd.", buildDate));
359         } finally {
360             httpWorker.logout();        	
361         }    
362         
363     }

365     /**
366      * Download the specified distributions, if necessary, extract resources
367      * from them, and use the results to create Maven artifacts as appropriate:
368      * <p>
369      * Try to install all of the main artifacts - e.g., those found in lib/*.jar
370      * and assembly/*.zip 
371      * <br>
372      * Try to match main artifacts to 'subartifacts' by name and attach them
373      * (with classifiers as necessary)
374      * 
375      * @param downloads
376      *            The list of licenses top be included in the distribution.
377      *            Multiple licenses are only allowed to support the inclusion of
378      *            optional modules.
379      * @param basedir
380      *            The directory into which results should be written
381      * @return A collection of Maven artifacts resulting from the download and
382      *         preparation of a supported Isomorphic SDK.
383      * @throws MojoExecutionException
384      *             When any fatal error occurs.
385      */
386     private Set<Module> collect(List<License> downloads, File basedir)
387         throws MojoExecutionException {
388 
389         File downloadTo = new File(basedir, "zip");
390         downloadTo.mkdirs();
391 
392         Downloads downloadManager = new Downloads(httpWorker);
393         downloadManager.setToFolder(downloadTo);
394         downloadManager.setOverwriteExistingFiles(overwrite);
395 
396         File[] existing = downloadTo.listFiles();
397         List<Distribution> distributions = new ArrayList<Distribution>();
398         try {
399             if (!skipDownload) {
400                 distributions.addAll(downloadManager.fetch(product, buildNumber, buildDate,
401                     downloads.toArray(new License[0])));
402             } else if (existing != null) {
403                 LOGGER.info("Creating local distribution from '{}'",
404                     downloadTo.getAbsolutePath());
405                 Distribution distribution = Distribution.get(product, license);
406                 distribution.getFiles().addAll(Arrays.asList(existing));
407                 distributions.add(distribution);
408             }
409 
410             if (!skipExtraction) {
411                 LOGGER.info("Unpacking downloaded file/s to '{}'", basedir);
412                 for (Distribution distribution : distributions) {
413                     distribution.unpack(basedir);
414                 }
415             }
416 
417             // it doesn't strictly read this way, but we're looking for
418             // lib/*.jar, pom/*.xml, assembly/*.zip
419             // TODO it'd be better if this didn't have to know where the files
420             // were located after unpacking
421             Collection<File> files = FileUtils.listFiles(
422                 basedir,
423                 FileFilterUtils.or(FileFilterUtils.suffixFileFilter("jar"),
424                     FileFilterUtils.suffixFileFilter("xml"),
425                     FileFilterUtils.suffixFileFilter("zip")),
426                 FileFilterUtils.or(FileFilterUtils.nameFileFilter("lib"),
427                     FileFilterUtils.nameFileFilter("pom"),
428                     FileFilterUtils.nameFileFilter("assembly"))
429             );
430             
431             if (files.isEmpty()) {
432                 throw new MojoExecutionException(
433                     String
434                         .format(
435                             "There don't appear to be any files to work with at '%s'.  Check earlier log entries for clues.",
436                             basedir.getAbsolutePath()));
437             }
438 
439             Set<Module> result = new TreeSet<Module>();
440             for (File file : files) {
441                 try {
442 
443                     String base = FilenameUtils.getBaseName(file.getName()
444                         .replaceAll("_", "-"));
445 
446                     // poms don't need anything else
447                     if ("xml".equals(FilenameUtils.getExtension(file.getName()))) {
448                         result.add(new Module(getModelFromFile(file)));
449                         continue;
450                     }
451 
452                     // for each jar/zip, find the matching pom
453                     IOFileFilter filter = new WildcardFileFilter(base + ".pom");
454                     Collection<File> poms = FileUtils.listFiles(basedir, filter,
455                         TrueFileFilter.INSTANCE);
456                     if (poms.size() != 1) {
457                         LOGGER
458                             .warn(
459                                 "Expected to find exactly 1 POM matching artifact with name '{}', but found {}.  Skpping installation.",
460                                 base, poms.size());
461                         continue;
462                     }
463 
464                     
465                     
466                     Model model = getModelFromFile(poms.iterator().next());
467                     Module module = new Module(model, file);
468 
469                     /*
470                      * Find the right javadoc bundle, matched on prefix. e.g.,
471                      * smartgwt-eval -> smartgwt-javadoc isomorphic-core-rpc ->
472                      * isomorphic-javadoc and add it to the main artifact with
473                      * the javadoc classifier. This seems appropriate as long as
474                      * a) there is no per-jar javadoc b) naming conventions are
475                      * adhered to (or can be corrected by plugin at extraction)
476                      */
477                     int index = base.indexOf("-");
478                     String prefix = base.substring(0, index);
479 
480                     Collection<File> doc = FileUtils.listFiles(new File(basedir, "doc"),
481                         FileFilterUtils.prefixFileFilter(prefix),
482                         FileFilterUtils.nameFileFilter("lib"));
483 
484                     if (doc.size() != 1) {
485                         LOGGER
486                             .debug(
487                                 "Found {} javadoc attachments with prefix '{}'.  Skipping attachment.",
488                                 doc.size(), prefix);
489                     } else {
490                         module.attach(doc.iterator().next(), "javadoc");
491                     }
492 
493                     result.add(module);
494                 } catch (ModelBuildingException e) {
495                     throw new MojoExecutionException("Error building model from POM", e);
496                 }
497             }
498             return result;
499         } catch (IOException e) {
500             throw new MojoExecutionException("Failure during assembly collection", e);
501         }
502     }

504     /**f
505      * Read the given POM so it can be used as the source of coordinates, etc.
506      * during artifact construction. Note that if this object's
507      * {@link #snapshots} property is true, and we're working with a development
508      * build ({@link #buildNumber} ends with 'd'), the POM is modified to remove
509      * the SNAPSHOT qualifier.
510      * 
511      * @param pom
512      *            the POM file containing the artifact metadata
513      * @return A Maven model to be used at
514      *         {@link com.isomorphic.maven.packaging.Module#Module(Model)}
515      *         Module construction
516      * @throws ModelBuildingException
517      *             if the Model cannot be built from the given POM
518      * @throws IOException
519      *             if the Model cannot be built from the given POM
520      */
521     private Model getModelFromFile(File pom) throws ModelBuildingException, IOException {
522 
523         if (buildNumber.endsWith("d") && !snapshots) {
524             LOGGER.info(
525                 "Rewriting file to remove SNAPSHOT qualifier from development POM '{}'",
526                 pom.getName());
527             String content = FileUtils.readFileToString(pom);
528             content = content.replaceAll("-SNAPSHOT", "");
529             FileUtils.write(pom, content);
530         }
531 
532         ProjectModelResolver resolver = new ProjectModelResolver(repositorySystemSession, null, repositorySystem, remoteRepositoryManager, project.getRemoteProjectRepositories(), RepositoryMerging.POM_DOMINANT, null);
533         ModelBuildingRequest request = new DefaultModelBuildingRequest();
534         request.setModelResolver(resolver);
535         request.setPomFile(pom);
536         
537         Model model = modelBuilder.buildRawModel(pom, 0, false).get();
538         Parent parent = model.getParent();
539         if (parent != null) {
540         		model.setGroupId(parent.getGroupId());
541         		model.setVersion(parent.getVersion());
542         }
543         
544         return model;
545     }
