100 	/**
101 	 * Decrypt settings and return the server element with the given id.  Useful for e.g., reading encrypted 
102 	 * user credentials.
103 	 * <p>
104 	 * Refer to http://maven.apache.org/guides/mini/guide-encryption.html
105 	 * 
106 	 * @param id the id of the server to be decrypted
107 	 * @return a Server with its protected elements decrypted, if one is found with the given id.  Null otherwise.
108 	 */
109     private Server getDecryptedServer(String id) { 
110         final SettingsDecryptionRequest settingsDecryptionRequest = new DefaultSettingsDecryptionRequest(); 
111         settingsDecryptionRequest.setServers(settings.getServers()); 
112         final SettingsDecryptionResult decrypt = settingsDecrypter.decrypt(settingsDecryptionRequest); 
113         List<Server> servers = decrypt.getServers();
114         
115         for (Server server : servers) {
116         	if (server.getId().equals(id)) {
117         		return server;
118         	}
119         }
120         return null;
121     }
