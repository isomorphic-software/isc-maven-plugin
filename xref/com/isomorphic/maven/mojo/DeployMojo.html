19  /**
20   * Deploys a collection of {@link Module}s to the Maven repository location indicated by the given {@link #repositoryUrl} property. 
21   * Functionally, pretty much just like the Deploy Plugin's deploy-file goal, except this one works on a collection.
22   * <p>
23   * Refer to http://maven.apache.org/plugins/maven-deploy-plugin/deploy-file-mojo.html
24   */
25  @Mojo(name="deploy", requiresProject=false)
26  public final class DeployMojo extends AbstractPackagerMojo {
27  	
28  	private static final Logger LOGGER = LoggerFactory.getLogger(DeployMojo.class);
29  	
30  	/**
31  	 * The identifier of a server entry from which Maven should read the authentication credentials
32  	 * to be used during deployment to {@link #repositoryUrl}. 
33  	 * <p>
34  	 * Refer to http://maven.apache.org/ref/3.1.1/maven-settings/settings.html#class_server
35  	 * @since 1.0.0
36  	 */
37  	@Parameter(property="repositoryId")
38  	private String repositoryId;
39  	
40  	/**
41  	 * The URL of the location to which the artifacts should be deployed.  e.g.,
42  	 * <p/>
43  	 * http://nexus.corp.int/nexus/content/repositories/thirdparty/
44  	 * 
45  	 * @since 1.0.0
46  	 */
47  	@Parameter(property="repositoryUrl", required=true)
48  	private String repositoryUrl;
49  	
50  	/**
51  	 * The repositoryType, as required by the Builder(String, String, String) constructor.
52  	 * <p>
53  	 * Refer to <a href="http://http://download.eclipse.org/aether/aether-core/0.9.0.M2/apidocs/org/eclipse/aether/repository/RemoteRepository.Builder.html#RemoteRepository.Builder(java.lang.String,%20java.lang.String,%20java.lang.String)">
54  	 * @since 1.0.0
55  	 */
56  	@Parameter(property="repositoryType", defaultValue="default")
57  	private String repositoryType;
58  	
59  	/**
60  	 * Deploy each of the provided {@link Module}s, along with their SubArtifacts (POMs, JavaDoc bundle, etc.), to the repository location
61  	 * indicated by {@link #repositoryUrl}.
62  	 */
63  	@Override 
64  	public void doExecute(Set<Module> artifacts) throws MojoExecutionException, MojoFailureException {
65  
66      	RemoteRepository.Builder builder = new RemoteRepository.Builder(repositoryId, repositoryType, repositoryUrl);
67  		Authentication authentication = getAuthentication(repositoryId);
68      	builder.setAuthentication(authentication);
69  
70      	LOGGER.info("Deploying to repositoryId '{}' with credentials: {}", repositoryId, authentication == null ? null : authentication.toString());
71      	
72      	RemoteRepository repository = builder.build();
73  		
74  		for (Module artifact : artifacts) {
75  	     
76  			DeployRequest deployRequest = new DeployRequest();
77  			deployRequest.addArtifact(artifact); 
78          	deployRequest.setRepository(repository);
79          	
80  			for (Artifact subArtifact : artifact.getAttachments()) {
81  	        	deployRequest.addArtifact(subArtifact);
82  	        }
83  	        
84  	        try {
85  				repositorySystem.deploy(repositorySystemSession, deployRequest);
86  			} catch (DeploymentException e) {
87  				throw new MojoFailureException("Deployment failed: ", e);
88  			}	
89  
90  		}		
91  	}
