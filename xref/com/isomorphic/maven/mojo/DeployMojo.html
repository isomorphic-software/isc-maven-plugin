19   * Deploys a collection of {@link Module}s to the Maven repository location indicated by the given {@link #repositoryUrl} property. 
20   * Functionally, pretty much just like the Deploy Plugin's deploy-file goal, except this one works on a collection.
21   * 
22   * @see http://maven.apache.org/plugins/maven-deploy-plugin/deploy-file-mojo.html
23   */
24  @Mojo(name="deploy", requiresProject=false)
25  public final class DeployMojo extends AbstractPackagerMojo {
26  	
27  	private static final Logger LOGGER = LoggerFactory.getLogger(DeployMojo.class);
28  	
29  	/**
30  	 * The identifier of a server entry from which Maven should read the authentication credentials
31  	 * to be used during deployment to {@link #repositoryUrl}. 
32  	 * 
33  	 * @see http://maven.apache.org/ref/3.1.1/maven-settings/settings.html#class_server
34  	 */
35  	@Parameter(property="repositoryId")
36  	private String repositoryId;
37  	
38  	/**
39  	 * The URL of the location to which the artifacts should be deployed.  e.g.,
40  	 * <p/>
41  	 * http://nexus.corp.int/nexus/content/repositories/thirdparty/
42  	 */
43  	@Parameter(property="repositoryUrl", required=true)
44  	private String repositoryUrl;
45  	
46  	/**
47  	 * The repositoryType, as required by the Builder(String, String, String) constructor.
48  	 * 
49  	 * @see <a href="http://http://download.eclipse.org/aether/aether-core/0.9.0.M2/apidocs/org/eclipse/aether/repository/RemoteRepository.Builder.html#RemoteRepository.Builder(java.lang.String,%20java.lang.String,%20java.lang.String)">
50  	 */
51  	@Parameter(property="repositoryType", defaultValue="default")
52  	private String repositoryType;
53  	
54  	/**
55  	 * Deploy each of the provided {@link Module}s, along with their SubArtifacts (POMs, JavaDoc bundle, etc.), to the repository location
56  	 * indicated by {@link #repositoryUrl}.
57  	 */
58  	@Override 
59  	public void doExecute(Set<Module> artifacts) throws MojoExecutionException, MojoFailureException {
60  	
61  		for (Module artifact : artifacts) {
62  	     
63  			DeployRequest deployRequest = new DeployRequest();
64  			deployRequest.addArtifact(artifact);
65  	        
66  	        for (Artifact subArtifact : artifact.getAttachments()) {
67  	        	deployRequest.addArtifact(subArtifact);
68  	        }
69  
70  	        try {
71  	        	RemoteRepository repository = new RemoteRepository.Builder(repositoryId, repositoryType, repositoryUrl).build();
72  	        	deployRequest.setRepository(repository);
73  				repositorySystem.deploy(repositorySystemSession, deployRequest);
74  			} catch (DeploymentException e) {
75  				throw new MojoFailureException("Deployment failed: ", e);
76  			}	
77  
78  		}		
79  	}
