18  /**
19   * Deploys a collection of {@link Module}s to the Maven repository location indicated by the given {@link #repositoryUrl} property. 
20   * Functionally, pretty much just like the Deploy Plugin's deploy-file goal, except this one works on a collection.
21   * <p>
22   * Refer to <a href="http://maven.apache.org/plugins/maven-deploy-plugin/deploy-file-mojo.html"></a>
23   */
24  @Mojo(name="deploy", requiresProject=false)
25  public final class DeployMojo extends AbstractPackagerMojo {
26  
27      private static final Logger LOGGER = LoggerFactory.getLogger(DeployMojo.class);
28  
29      /**
30       * The identifier of a server entry from which Maven should read the authentication credentials
31       * to be used during deployment to {@link #repositoryUrl}.
32       * <p>
33       * Refer to <a href="http://maven.apache.org/ref/3.1.1/maven-settings/settings.html#class_server"></a>
34       * @since 1.0.0
35       */
36      @Parameter(property="repositoryId")
37      private String repositoryId;
38  
39      /**
40       * The URL of the location to which the artifacts should be deployed.  e.g.,
41       * <p/>
42       * <a href="http://nexus.corp.int/nexus/content/repositories/thirdparty/"></a>
43       *
44       * @since 1.0.0
45       */
46      @Parameter(property="repositoryUrl", required=true)
47      private String repositoryUrl;
48  
49      /**
50       * The repositoryType, as required by the Builder(String, String, String) constructor.
51       * <p>
52       * Refer to <a href="http://http://download.eclipse.org/aether/aether-core/0.9.0.M2/apidocs/org/eclipse/aether/repository/RemoteRepository.Builder.html#RemoteRepository.Builder(java.lang.String,%20java.lang.String,%20java.lang.String)">
53       * @since 1.0.0
54       */
55      @Parameter(property="repositoryType", defaultValue="default")
56      private String repositoryType;
57  
58      /**
59       * Deploy each of the provided {@link Module}s, along with their SubArtifacts (POMs, JavaDoc bundle, etc.), to the repository location
60       * indicated by {@link #repositoryUrl}.
61       */
62      @Override
63      public void doExecute(Set<Module> artifacts) throws MojoExecutionException, MojoFailureException {
64  
65          RemoteRepository.Builder builder = new RemoteRepository.Builder(repositoryId, repositoryType, repositoryUrl);
66          Authentication authentication = getAuthentication(repositoryId);
67          builder.setAuthentication(authentication);
68  
69          LOGGER.info("Deploying to repositoryId '{}' with credentials: {}", repositoryId, authentication == null ? null : authentication.toString());
70  
71          RemoteRepository repository = builder.build();
72  
73          for (Module artifact : artifacts) {
74  
75              DeployRequest deployRequest = new DeployRequest();
76              deployRequest.addArtifact(artifact);
77              deployRequest.setRepository(repository);
78  
79              for (Artifact subArtifact : artifact.getAttachments()) {
80                  deployRequest.addArtifact(subArtifact);
81              }
82  
83              try {
84                  repositorySystem.deploy(repositorySystemSession, deployRequest);
85              } catch (DeploymentException e) {
86                  throw new MojoFailureException("Deployment failed: ", e);
87              }
88  
89          }
90      }
91  
92  }
