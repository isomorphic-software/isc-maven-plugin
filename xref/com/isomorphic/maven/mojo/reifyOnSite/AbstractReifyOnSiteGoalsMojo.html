43  /**
44   * This is the shared Reify OnSite Mojo that the goals install-reify-onsite and
45   * upgrade-reify-onsite inherit from.  It was introduced with those goals in order to provide
46   * a common ancestor for documenting parameters, because many of the parameters that were
47   * previously annotated and documented in AbstractBaseMojo are not applicable to these new
48   * Reify OnSite goals, but we do want those goals to extend AbstractBaseMojo because they
49   * make use of substantial shared facilities, like obtaining the latest release number,
50   * downloading and unpacking.
51   *
52   * So we define the common annotations and docs for the Reify OnSite functions here, so they
53   * are inherited by the Reify OnSite goals but not the existing "core" goals like "install"
54   */
55  public class AbstractReifyOnSiteGoalsMojo extends AbstractPackagerMojo {
56  
57      protected Logger log = LoggerFactory.getLogger(AbstractReifyOnSiteGoalsMojo.class);

60      /**
61       * Technically the product should be "REIFY_ONSITE", but there is no need to specify it
62       * (and indeed it will be ignored if you do).  REIFY_ONSITE is implied for the Reify
63       * OnSite goals, so we force it to that setting
64       *
65       * @since 1.0.0
66       */
67      @Parameter(property = "product", defaultValue = "REIFY_ONSITE")
68      public void setProduct(Product product) {
69          this.product = product;
70      }
71  
72      /**
73       * Technically the license should be "ENTERPRISE", but there is no need to specify it
74       * (and indeed it will be ignored if you do).  ENTERPRISE is implied for the Reify OnSite
75       * goals, so we force it to that setting
76       *
77       * @since 1.0.0
78       */
79      @Parameter(property = "license", defaultValue = "ENTERPRISE")
80      public void setLicense(License license) {
81          this.license = license;
82      }

89      @Override
90      public void execute() throws MojoExecutionException, MojoFailureException {
91          if (product != Product.REIFY_ONSITE) {
92              log.info("For goal '" + mojoExecution.getGoal() + "', product MUST be REIFY_ONSITE.  Overriding and continuing");
93              product = Product.REIFY_ONSITE;
94          }
95          if (license != License.ENTERPRISE) {
96              log.info("For goal '" + mojoExecution.getGoal() + "', license MUST be ENTERPRISE.  Overriding and continuing");
97              license = License.ENTERPRISE;
98          }
99          super.execute();
100     }

102     @Override
103     public void doExecute(Set<Module> artifacts) throws MojoExecutionException, MojoFailureException {
104         throw new MojoExecutionException("The doExecute() override that accepts a Set of Modules is not supported for " +
105                 "goal " + mojoExecution.getGoal() + "; this goal is about deploying " +
106                 "Reify OnSite, not installing Maven-managed assets");
107     }

109     @Override
110     protected void skin(File basedir, Map<String, String> skinResources) throws MojoExecutionException, IOException {
111 
112         if (skins == null && mojoExecution.getGoal().equals("install-reify-onsite")) {
113             // Install all skins if no "skins" property is provided
114             return;
115         }
116 
117         // Otherwise, we install the list of skins provided (plus some standard ones).  If we
118         // are upgrading, we only upgrade the installed skins, plus any that the user specified
119         // in the skins param
120         log.info("Pruning unwanted skins...");
121 
122         // Reify OnSite goals do not need to extract or unJar anything, so the "skinResources"
123         // is kinda redundant.  We just use it to store the path to the skins/ directory
124         // against a known key
125 
126         List<String> requested;
127         if (skins != null) {
128             requested = new ArrayList<>(Splitter.on(",").trimResults().splitToList(skins.toLowerCase()));
129         } else {
130             requested = new ArrayList<>();
131         }
132 
133         String subdir = "reifyOnSite";
134         String skinDirName = subdir + "/" + skinResources.get("REIFY_ONSITE");
135         File skinDir = new File(basedir, skinDirName);
136 
137         // The Install goal adds the same set of standard skins that a standard framework
138         // goal adds; the Upgrade goal also adds any skins that are currently installed
139         List<String> additionalSkins = getAdditionalSkins(skinResources.get("REIFY_ONSITE"));
140         requested.addAll(additionalSkins);
141 
142         String[] skinNames = skinDir.list();
143         if (skinNames != null && skinNames.length > 0) {
144             for (String skinName : skinNames) {
145                 if (requested.contains(skinName.toLowerCase())) continue;
146                 File dirToRemove = new File(skinDir, skinName);
147                 log.info("Deleting '{}' skin resources at '{}'", skinName, dirToRemove.getAbsolutePath());
148                 if (FileSystemUtils.deleteRecursively(dirToRemove)) {
149                     log.info("Deleting '{}' skin resources at '{}' completed successfully", skinName, dirToRemove.getAbsolutePath());
150                 } else {
151                     log.info("Deleting '{}' skin resources at '{}' encountered problems, some skin resources may remain", skinName, dirToRemove.getAbsolutePath());
152                 }
153             }
154         } else {
155             log.info("No 'skins' directory found in download, skipping");
156         }
157     }
158 
159     protected List<String> getAdditionalSkins(String skinDirName) {
160         List<String> additional = new ArrayList<String>();
161         // preserve for simplicity
162         additional.add("enterprise");
163         additional.add("fonts");
164         additional.add("toolskin");
165         additional.add("toolskinnative");
166 
167         return additional;
168     }

171     protected File sourceRoot, targetRoot;
172     protected int fileCount;
173 
174     protected void copyRecursively(File file, AntPathMatcherFilter filter)
175             throws MojoExecutionException, MojoFailureException
176     {
177         String relPath = sourceRoot.toPath().relativize(file.toPath()).toString()
178                 .replace(File.separatorChar, '/'); // normalize to forward slashes in case we're running on Windows
179 
180         if (file.isDirectory()) {
181             // Skip excluded directories entirely
182             if (!relPath.isEmpty() && !filter.accept(relPath + "/")) {
183                 return;
184             }
185             File[] children = file.listFiles();
186             if (children != null) {
187                 for (File child : children) {
188                     copyRecursively(child, filter);
189                 }
190             }
191         } else {
192             if (filter.accept(relPath)) {
193                 fileCount++;
194                 try {
195                     copyFile(file, new File(targetRoot, relPath));
196                 } catch(IOException ioe) {
197                     throw new MojoFailureException("Caught excpetion trying to copy files", ioe);
198                 }
199             }
200         }
201     }
202 
203     protected void copyFile(File src, File dst) throws IOException {
204         if (!dst.getParentFile().exists()) {
205             dst.getParentFile().mkdirs();
206         }
207         try (InputStream in = new FileInputStream(src);
208              OutputStream out = new FileOutputStream(dst)) {
209             byte[] buf = new byte[8192];
210             int len;
211             while ((len = in.read(buf)) != -1) {
212                 out.write(buf, 0, len);
213             }
214         }
215     }
