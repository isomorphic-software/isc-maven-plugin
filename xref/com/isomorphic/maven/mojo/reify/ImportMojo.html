52  /**
53   * Provides for single-step download and extraction of assets hosted on the Reify platform.
54   * While there is nothing to prevent a user from modifying the imported resources, changes 
55   * should almost always be made using Reify and then re-imported.
56   * <p>
57   * If necessary, use the API (e.g., RPCManager.createScreen, Canvas.getByLocalId) to modify 
58   * imported screen definitions at runtime.
59   * <p>
60   * Imported MockDataSources can be replaced using e.g., SQLDataSources having the same ID in
61   * another directory - there is no reason to modify the contents of the 'mock' subdirectory.
62   * <p> 
63   * To encourage recommended usage, the reify-import goal takes steps to detect local changes 
64   * and fail when any are found.  Refer to the {@link #skipOverwriteProtection}
65   * parameter for details.
66   */
67  // Set requiresProject: false so that we can run the whole thing from an Ant project w/ no POM
68  @Mojo(name="reify-import", requiresProject=false)
69  public class ImportMojo extends AbstractBaseMojo {

136     /**
137      * When a project is imported, its screen and datasource assets are marked with a checksum
138      * representing the contents of each file.  Subsequent imports are preceded by a check to 
139      * make sure each of these checksums still match the file's content.  In the event that
140      * any do not, the import fails by default.

176 	 * and installs Reify project contents to the local project environment.

178 	@Override
179 	public void execute() throws MojoExecutionException, MojoFailureException {
180 
181         try {
182 
183     		// derive filenames from project name
184     		zipFileName = projectName + ".proj.zip";
185     		projectFileName = projectName + ".proj.xml";
186         	
187 			// check to make sure there are no local changes, unless configured otherwise
188 			if (! skipOverwriteProtection) {
189 				checkForLocalModifications();	
190 			}
191 			
192 			// allow explicit configuration from ImportTask (for Ant builds)
193 			if (credentials == null) {
194 		        credentials = getCredentials(serverId);			
195 			}
196 	        if (credentials == null) {
197 	        	throw new MojoExecutionException("Reify credentials have not been configured.  Check Maven settings for the '" + serverId + "' server.");
198 	        }
199 	
200 	        // allow for manual construction of the Proxy (again, for Ant builds)
201 	        httpWorker = new HttpRequestManager(host, credentials, settings != null ? settings.getActiveProxy() : proxy);
202 	        httpWorker.login();
203 	        
204 	        // get fresh project metadata and archive from the server
205 	    	Document project = downloadProjectDocument();
206 	    	File archive = downloadProjectArchive(project);
207 
208 			LOGGER.info("Importing Reify project assets from download at '{}'...", archive.getCanonicalPath());
209 			File unpacked = new File(workdir, projectName);
210 			ArchiveUtils.unzip(archive, unpacked);
211 
212 			// update the new project file with checksums of file content for overwrite protection on future imports
213 			writeProjectFile(project, unpacked);
214 
215 			// and copy the result to application source
216 			FileUtils.copyDirectory(unpacked, webappDir);
217 			
218         } catch (MojoExecutionException e) {
219         	throw e;
220 		} catch (Exception e) {
221 			throw new MojoExecutionException("Import Error", e);
222 		}  finally {
223 			try {
224 				httpWorker.logout();
225 			} catch (Exception ignore) {}
226 		}

229 	/**
230 	 * Checks for local modifications, as documented at {@link #skipOverwriteProtection}},
231 	 * and fails when any local changes are detected.
232 	 * <p>
233 	 * Note that files are expected in the configured locations - screens in {@link #uiDir}
234 	 * and datasources in {@link #datasourcesDir} (or the {@link #datasourcesDir}/mock 
235 	 * subdirectory).
236 	 * 
237  	 * @throws MojoExecutionException when any local changes are detected
238  	 * @throws Exception on any other error
239 	 */
240 	private void checkForLocalModifications() throws MojoExecutionException, Exception {
241 		
242 		File projectFile = new File(webappDir, projectFileDir + "/" + projectFileName);
243 		if (! projectFile.exists()) {
244 			LOGGER.info("Project file does not exist at '{}', but will be created during import.", projectFile.getCanonicalPath());
245 			return;
246 		}
247 
248 		SAXReader reader = new SAXReader();
249         Document document = reader.read(projectFile);
250 
251 		String msg = "File '%s' appears to have been modified since the last import.  Aborting to provide an opprtunity to investigate.  \n" + 
252 				"You can either revert your changes, delete the file, or remove its checksum from the project file to continue. \n" + 
253 				"You can also disable the check for this and all other files using the skipOverwriteProtection parameter.";
254         
255         List<Node> screens = getScreenNodes(document);
256         for (Node screen : screens) {
257         	String expected = screen.valueOf("checksum");
258         	if (expected.isEmpty()) {
259         		continue;
260         	}
261         	File file = getScreen(screen, webappDir);
262         	if (! file.exists()) {
263         		continue;
264         	}
265         	
266         	String actual = Files.asByteSource(file).hash(Hashing.sha256()).toString(); 
267         	if (!actual.equals(expected)) {
268         		LOGGER.debug("Checksum mismatch - Expected: '{}' Actual: '{}'", expected, actual);
269         		throw new MojoExecutionException(String.format(msg, file.getCanonicalPath()));
270         	}	        	
271 		}
272         List<Node> datasources = getDataSourceNodes(document);
273         for (Node ds : datasources) {
274         	String expected = ds.valueOf("checksum");
275         	if (expected.isEmpty()) {
276         		continue;
277         	}
278         	File file = getDataSource(ds, webappDir);
279         	if (! file.exists()) {
280         		continue;
281         	}
282         	
283         	String actual = Files.asByteSource(file).hash(Hashing.sha256()).toString(); 
284         	if (!actual.equals(expected)) {
285         		LOGGER.debug("Checksum mismatch - Expected: '{}' Actual: '{}'", expected, actual);
286         		throw new MojoExecutionException(String.format(msg, file.getCanonicalPath()));
287         	}	        	
288 		}

292 	 * Requests the named project's metadata from the reify server.
293 	 * 
294 	 * @return Document containing the result.
295 	 * @throws Exception when any error occurs
296 	 */
297 	private Document downloadProjectDocument() throws Exception {
298 		
299         LOGGER.info("Contacting server for Reify project metadata...");
300         
301         HttpGet request = new HttpGet("/isomorphic/RESTHandler/hostedProjects?fileName=" + URLEncoder.encode(projectName) + "&isc_dataFormat=xml");
302         HttpResponse response = httpWorker.execute(request);
303 
304         String body = null;
305         try {
306 	        
307         	body = EntityUtils.toString(response.getEntity());
308 	   		Document metadata =  DocumentHelper.parseText(body);
309 	   		
310 			Node fileContents = metadata.selectSingleNode("//fileContents");    	
311 	    	String project = fileContents.getText();
312 			
313 	    	return DocumentHelper.parseText(project);
314 	    	
315         } catch (Exception e) {
316         	
317     		LOGGER.error("Response from server:" + System.getProperty("line.separator") + body);
318     		throw new Exception(String.format("Unexpected response to request for project file.  Check that user '%s' is able to access the project named '%s'", 
319     				credentials.getUserName(), projectName));
320 		}
321 	}

323 	/**
324 	 * Requests the named project's export archive, including all screens and all datasources,
325 	 * downloads it, and saves it to {@link #workdir}.
326 	 * 
327 	 * @param project metadata used to provide the names of requested assets
328 	 * @return zip file containing the export result
329 	 * @throws Exception when any error occurs
330 	 */
331 	private File downloadProjectArchive(Document project) throws Exception {
332         
333 		Map<String, Object> context = new HashMap<String, Object>();
334     	
335     	context.put("datasourcesDir", datasourcesDir);
336     	context.put("uiDir", uiDir);
337     	context.put("projectFileDir", projectFileDir);
338     	context.put("projectFileName", projectFileName);
339     	context.put("zipFileName", zipFileName);
340     	
341 		context.put("screens", getScreenNames(project));
342 		context.put("datasources", getDataSourceNames(project));
343 
344     	StringWriter writer = new StringWriter();
345 		freemarkerConfig.getTemplate("transaction.ftl").process(context, writer);
346 
347 		List<NameValuePair> params = new ArrayList<NameValuePair>();
348 	    params.add(new BasicNameValuePair("_transaction", writer.toString()));
349 	    
350 		HttpPost request = new HttpPost("/isomorphic/IDACall?isc_rpc=1");
351 	    request.setEntity(new UrlEncodedFormEntity(params));
352 		
353         LOGGER.info("Downloading '{}' project export from '{}'...", projectName, host.getHostName());
354 	    
355 		HttpResponse response = httpWorker.execute(request);
356 		HttpEntity entity = response.getEntity();
357 		
358     	if (! "application/zip".equals(entity.getContentType().getValue())) {
359     		throw new MojoExecutionException("Response does not contain zip file contents:\n" + EntityUtils.toString(entity));
360     	}
361 
362     	FileUtils.forceMkdir(workdir);
363     	FileUtils.cleanDirectory(workdir);
364 		File zip = new File(workdir + "/" + zipFileName);
365 
366 		OutputStream outputStream = null;
367         try {
368             outputStream = new LoggingCountingOutputStream(new FileOutputStream(zip), entity.getContentLength());
369             entity.writeTo(outputStream);
370         } catch (Exception e) {
371             throw new MojoExecutionException("Error writing file to '" + zip.getAbsolutePath() + "'", e);
372         } finally {
373             IOUtils.closeQuietly(outputStream);
374         }
375 		
376         return zip;
377 	}

379 	/**
380 	 * Calculates checksums for every file in the downloaded archive, writes them to
381 	 * the given project file, and stores the result on disk at a folder named for 
382 	 * {@link #projectFileDir}, relative to the given parent.
383 	 * 
384 	 * @param project metadata used to provide the names of requested assets
385 	 * @param parent the parent directory
386 	 * @throws Exception whenany error occurs
387 	 */
388 	private void writeProjectFile(Document project, File parent) throws Exception {
389 
390 		for (Node screen : getScreenNodes(project)) {
391 
392         	File file = getScreen(screen, parent);
393         	String checksum = Files.asByteSource(file).hash(Hashing.sha256()).toString();
394 
395         	Element element = ((Element) screen).addElement("checksum");
396         	element.setText(checksum);			
397 		}
398 		for (Node datasource : getDataSourceNodes(project)) {
399 			
400 			File file = getDataSource(datasource, parent);
401         	String checksum = Files.asByteSource(file).hash(Hashing.sha256()).toString();
402 
403         	Element element = ((Element) datasource).addElement("checksum");
404         	element.setText(checksum);
405 		}
406 		
407 		// pretty print the file with changes
408 		File projectFile = FileUtils.getFile(parent, projectFileDir, projectFileName);
409 		OutputFormat purdy = OutputFormat.createPrettyPrint();
410 		XMLWriter writer = new XMLWriter(new FileWriter(projectFile), purdy);
411 
412 		writer.write(project);
413 		writer.flush();
414 		writer.close();
415 	}

456 
457 		String name = getDataSourceName(metadata);
458     	
459 		// Check 'mock' subdirectory first, fallback to datasourcesDir if not found.
460 		File file = FileUtils.getFile(parent, datasourcesDir, "mock", name + ".ds.xml");
461     	if (! file.exists()) {
462         	file = FileUtils.getFile(parent, datasourcesDir, name + ".ds.xml");
463     	}
