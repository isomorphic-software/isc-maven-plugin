69  /**
70   * Provides for single-step download and extraction of assets hosted on the Reify platform.
71   * While there is nothing to prevent a user from modifying the imported resources, changes 
72   * should almost always be made using Reify and then re-imported. If necessary, use the API 
73   * (e.g., RPCManager.createScreen, Canvas.getByLocalId) to modify imported screen definitions 
74   * at runtime.  Imported MockDataSources can simply be replaced using e.g., SQLDataSources 
75   * having the same ID in another directory - there is no reason to modify the contents of the 
76   * 'mock' subdirectory.
77   * <p> 
78   * To encourage recommended usage, the reify-import goal takes steps to detect local changes 
79   * and fail when any are found.  Refer to the {@link #skipOverwriteProtection}
80   * parameter for details.  Further, a validation step (optional, see {@link ImportMojo#skipValidationOnImport}) 
81   * attempts to detect <a href="https://www.smartclient.com/smartgwtee-latest/server/javadoc/com/isomorphic/tools/ReifyDataSourceValidator.html">commonly found discrepancies</a> 
82   * between your MockDataSources and working DataSources.
83   * <p>
84   * The <a href="https://www.smartclient.com/smartclient-latest/isomorphic/system/reference/?id=group..reifyForDevelopers">Reify for Developers</a> 
85   * documentation topic contains further discussion around best practices during the design / development cycle.
86   * <p>
87   * If you've built your project using one of the Maven archetypes for either  
88   * <a href="https://www.smartclient.com/smartgwt-latest/javadoc/com/smartgwt/client/docs/MavenSupport.html">SmartGWT</a>
89   * or
90   * <a href="https://www.smartclient.com/smartclient-latest/isomorphic/system/reference/?id=group..mavenSupport">SmartClient</a>, 
91   * you should have a skeleton configuration in you POM already.  if not, add something like the following:
92   *
93   * <pre>
94   * &lt;pluginManagement&gt;
95   *     &lt;plugins&gt;
96   *         &lt;plugin&gt;
97   *             &lt;groupId&gt;com.isomorphic&lt;/groupId&gt;
98   *             &lt;artifactId&gt;isc-maven-plugin&lt;/artifactId&gt;
99   *             &lt;version&gt;1.4.0-SNAPSHOT&lt;/version&gt;
100  *             &lt;!-- the m2pluginextras dependency will be required when skipValidationOnImport = false --&gt;
101  *             &lt;dependencies&gt;
102  *                 &lt;dependency&gt;
103  *                     &lt;groupId&gt;com.isomorphic.extras&lt;/groupId&gt;
104  *                     &lt;artifactId&gt;isomorphic-m2pluginextras&lt;/artifactId&gt;
105  *                    &lt;version&gt;${smartgwt.version}&lt;/version&gt;
106  *                 &lt;/dependency&gt;       
107  *             &lt;/dependencies&gt;
108  *         &lt;/plugin&gt;
109  *     &lt;/plugins&gt;
110  * &lt;/pluginManagement&gt;                
111  * 
112  * &lt;build&gt;
113  *     &lt;plugins&gt;
114  *       &lt;plugin&gt;
115  *           &lt;groupId&gt;com.isomorphic&lt;/groupId&gt;
116  *           &lt;artifactId&gt;isc-maven-plugin&lt;/artifactId&gt;
117  *           &lt;configuration&gt;
118  *             &lt;smartclientRuntimeDir&gt;${project.parent.build.directory}/gwt/launcherDir/myapplication/sc&lt;/smartclientRuntimeDir&gt;
119  *             &lt;dataSourcesDir&gt;WEB-INF/ds/classic-models&lt;/dataSourcesDir&gt;
120  *           &lt;/configuration&gt;
121  *       &lt;/plugin&gt;
122  *     &lt;/plugins&gt;
123  * &lt;/build&gt;
124  * </pre> 
125  * and check that the SmartClient runtime has been extracted to the configured location 
126  * (via e.g., mvn war:exploded, mvn jetty:run).
127  */

348     /**
349      * After ensuring there are no local modifications (unless otherwise configured), downloads
350      * and installs Reify project contents to the local project environment.
351      */
352     @Override
353     public void execute() throws MojoExecutionException, MojoFailureException {
354 
355         try {
356 
357             // derive filenames from project name
358             zipFileName = projectName + ".proj.zip";
359             projectFileName = projectName + ".proj.xml";
360 
361             // check to make sure there are no local changes, unless configured otherwise
362             if (! skipOverwriteProtection) {
363                 checkForLocalModifications();
364             }
365 
366             // allow explicit configuration from ImportTask (for Ant builds)
367             if (credentials == null) {
368                 credentials = getCredentials(serverId);
369             }
370             if (credentials == null) {
371                 throw new MojoExecutionException("Reify credentials have not been configured.  Check Maven settings for the '" + serverId + "' server.");
372             }
373 
374             // allow for manual construction of the Proxy (again, for Ant builds)
375             httpWorker = new HttpRequestManager(host, credentials, settings != null ? settings.getActiveProxy() : proxy);
376             httpWorker.login();
377 
378             // get fresh project metadata and archive from the server
379             Document project = downloadProjectDocument();
380             File archive = downloadProjectArchive(project);
381 
382             getLog().info(String.format("Importing Reify project assets from download at '%s'...", archive.getCanonicalPath()));
383             File unpacked = new File(workdir, projectName);
384             ArchiveUtils.unzip(archive, unpacked);
385 
386             // update the new project file with checksums of file content for overwrite protection on future imports
387             writeProjectFile(project, unpacked);
388 
389             // and copy the result to application source
390             FileUtils.copyDirectory(unpacked, webappDir);
391 
392             // finally, if the user has for some reason decided they want us to mangle their source, do so
393             if (modifyWelcomeFiles || drawOnWelcomeFiles) {
394                 modifyWelcomeFiles();
395             }
396 
397             // and validate whatever live datasources exist against the newly imported mocks
398             if (! skipValidationOnImport) {
399                 getLog().info("Validating MockDataSources against any working DataSources...");
400                 validate();
401             }
402 
403         } catch (MojoExecutionException e) {
404             throw e;
405         } catch (Exception e) {
406             throw new MojoExecutionException("Import Error", e);
407         }  finally {
408             try {
409                 httpWorker.logout();
410             } catch (Exception ignore) {}
411         }
412     }

414     /**
415      * Compares mock DataSources to live, logging findings and throwing an Exception if and
416      * when the {@link #validationFailureThreshold} is exceeded.
417      */
418     protected void validate() {
419 
420         try {
421 
422             ReifyDataSourceValidator validator = new ReifyDataSourceValidator(smartclientRuntimeDir.getCanonicalPath(), webappDir + "/" + dataSourcesDir);
423             validator.setMockDataSourcesBasePath(webappDir + "/" + mockDataSourcesDir);
424 
425             List<ErrorReport> result = validator.verify();
426             boolean fail = false;
427             if (! result.isEmpty()) {
428                 for (ErrorReport report : result) {
429                     for (String key : report.keySet()) {
430                         List<ErrorMessage> errors = report.getErrors(key);
431 
432                         for (ErrorMessage msg : errors) {
433 
434                             String output = null;
435                             if (key.equals(report.getDataSourceId())) {
436                                 output = String.format("%s: %s", report.getDataSourceId(), msg.getErrorString());
437                             } else {
438                                 output = String.format("%s.%s: %s", report.getDataSourceId(), key, msg.getErrorString());
439                             }
440 
441                             Severity severity = msg.getSeverity();
442                             if (severity.compareTo(validationFailureThreshold) >= 0) {
443                                 fail = true;
444                             }
445 
446                             if (severity == INFO) {
447                                 getLog().info(output);
448                             }
449                             if (msg.getSeverity() == WARN) {
450                                 getLog().warn(output);
451                             }
452                             if (msg.getSeverity() == ERROR) {
453                                 getLog().error(output);
454                             }
455                         }
456                     }
457                 }
458             } else {
459                 getLog().info("No validation errors were found.");
460             }
461             if (fail) {
462                 throw new MojoExecutionException("Mock DataSource validation failure.  Please refer to ReifyDataSourceValidator log output for detailed report.");
463             }
464         } catch (Exception e) {
465             throw new RuntimeException(e);
466         }
467     }

469     /**
470      * Checks for local modifications, as documented at {@link #skipOverwriteProtection}},
471      * and fails when any local changes are detected.
472      * <p>
473      * Note that files are expected in the configured locations - screens in {@link #uiDir}
474      * and datasources in {@link #mockDataSourcesDir} / {@link #dataSourcesDir}
475      * subdirectories).
476      *
477      * @throws MojoExecutionException when any local changes are detected
478      * @throws Exception on any other error
479      */
480     private void checkForLocalModifications() throws MojoExecutionException, Exception {
481 
482         File projectFile = new File(webappDir, projectFileDir + "/" + projectFileName);
483         if (! projectFile.exists()) {
484             //LOGGER.info("Project file does not exist at '{}', but will be created during import.", projectFile.getCanonicalPath());
485             getLog().info(String.format("Project file does not exist at '%s', but will be created during import.", projectFile.getCanonicalPath()));
486             return;
487         }
488 
489         SAXReader reader = new SAXReader();
490         Document document = reader.read(projectFile);
491 
492         String msg = "File '%s' appears to have been modified since the last import.  Aborting to provide an opprtunity to investigate.  \n" +
493                 "You can either revert your changes, delete the file, or remove its checksum from the project file to continue. \n" +
494                 "You can also disable the check for this and all other files using the skipOverwriteProtection parameter.";
495         
496         List<Node> screens = getScreenNodes(document);
497         for (Node screen : screens) {
498             String expected = screen.valueOf("checksum");
499             if (expected.isEmpty()) {
500                 continue;
501             }
502             File file = getScreen(screen, webappDir);
503             if (! file.exists()) {
504                 continue;
505             }
506 
507             String actual = Files.asByteSource(file).hash(Hashing.sha256()).toString();
508             if (!actual.equals(expected)) {
509                 getLog().debug(String.format("Checksum mismatch - Expected: '%s' Actual: '%s'", expected, actual));
510                 throw new MojoExecutionException(String.format(msg, file.getCanonicalPath()));
511             }
512         }
513         List<Node> datasources = getDataSourceNodes(document);
514         for (Node ds : datasources) {
515             String expected = ds.valueOf("checksum");
516             if (expected.isEmpty()) {
517                 continue;
518             }
519             File file = getDataSource(ds, webappDir);
520             if (! file.exists()) {
521                 continue;
522             }
523 
524             String actual = Files.asByteSource(file).hash(Hashing.sha256()).toString();
525             if (!actual.equals(expected)) {
526                 getLog().debug(String.format("Checksum mismatch - Expected: '%s' Actual: '%s'", expected, actual));
527                 throw new MojoExecutionException(String.format(msg, file.getCanonicalPath()));
528             }
529         }
530     }

532     /**
533      * Requests the named project's metadata from the reify server.
534      *
535      * @return Document containing the result.
536      * @throws Exception when any error occurs
537      */
538     private Document downloadProjectDocument() throws Exception {
539 
540         getLog().info("Contacting server for Reify project metadata...");
541         
542         HttpGet request = new HttpGet("/isomorphic/RESTHandler/hostedProjects?fileName=" + URLEncoder.encode(projectName) + "&isc_dataFormat=xml");
543         HttpResponse response = httpWorker.execute(request);
544 
545         String body = null;
546         try {
547 
548             body = EntityUtils.toString(response.getEntity());
549             Document metadata =  DocumentHelper.parseText(body);
550 
551             Node fileContents = metadata.selectSingleNode("//fileContents");
552             String project = fileContents.getText();
553 
554             return DocumentHelper.parseText(project);
555 
556         } catch (Exception e) {
557 
558             getLog().error("Response from server:" + System.getProperty("line.separator") + body);
559             throw new Exception(String.format("Unexpected response to request for project file.  Check that user '%s' is able to access the project named '%s'",
560                     credentials.getUserName(), projectName));
561         }
562     }

565         // SmartGWT builds have their modules directory directly under $smartclientRuntimeDir,
566         // as opposed to SmartClient, which expects it at $smartclientRuntimeDir/system/modules

570     private String getIsomorphicDirPath(String relativeToPath) {
571 
572         String[] parentDirectories = relativeToPath.split("/");
573         int parentDirectoryCount = parentDirectories == null ? 0 : parentDirectories.length;
574         String prefix = "";
575         for (int i=1; i < parentDirectoryCount; i++) {
576             prefix = "../" + prefix;
577         }
578 
579         String name = null;
580         // e.g., myapplication/sc vs isomorphic
581         if (isSmartGWTBuild()) {
582             name = smartclientRuntimeDir.getParentFile().getName() + "/" + smartclientRuntimeDir.getName();
583         } else {
584             name = smartclientRuntimeDir.getName();
585         }
586         return prefix + name;
587     }

589     /**
590      * Requests the named project's export archive, including all screens and all datasources,
591      * downloads it, and saves it to {@link #workdir}.
592      *
593      * @param project metadata used to provide the names of requested assets
594      * @return zip file containing the export result
595      * @throws Exception when any error occurs
596      */
597     private File downloadProjectArchive(Document project) throws Exception {
598         
599         Map<String, Object> context = new HashMap<String, Object>();
600         context.put("projectName", projectName);
601         context.put("datasourcesDir", dataSourcesDir);
602         context.put("mockDatasourcesDir", mockDataSourcesDir);
603         context.put("uiDir", uiDir);
604         context.put("projectFileDir", projectFileDir);
605         context.put("projectFileName", projectFileName);
606         context.put("zipFileName", zipFileName);
607 
608         context.put("screens", getScreenNames(project));
609         context.put("datasources", getDataSourceNames(project));
610 
611         context.put("includeTestJsp",  includeTestJsp);
612         context.put("testJspPathname", testJspPathname);
613         StringWriter jspWriter  = new StringWriter();

698     private void modifyWelcomeFiles() throws Exception {
699 
700         File welcomeFilesConfig = new File(webappDir, "WEB-INF/web.xml");
701         if (! welcomeFilesConfig.exists()) {
702             String msg = String.format("No web.xml found at '%s'", welcomeFilesConfig.getCanonicalPath());
703             throw new RuntimeException(msg);

706         List<File> welcomeFiles = new ArrayList<File>();
707 
708         SAXReader reader = new SAXReader();
709         Document webXml = reader.read(welcomeFilesConfig);
710         
711         XPath xpath = webXml.createXPath("//jee:welcome-file");
712         xpath.setNamespaceURIs(ImmutableMap.of("jee", "http://java.sun.com/xml/ns/javaee"));
713         List<Node> list = xpath.selectNodes(webXml);
714         for (Node node : list) {
715             String val = node.getStringValue();
716             File file = FileUtils.getFile(webappDir, val);
717             if (! file.exists()) {
718                 getLog().warn(String.format("Welcome file '{}' not found.  Skipping.", val));
719                 continue;
720             }
721             welcomeFiles.add(file);
722         }
723         if (welcomeFiles.isEmpty()) {
724             File jsp = FileUtils.getFile(webappDir, "index.jsp");
725             File html = FileUtils.getFile(webappDir, "index.html");
726             if (jsp.exists()) {
727                 welcomeFiles.add(jsp);
728             }
729             if (html.exists()) {
730                 welcomeFiles.add(html);
731             }
732         }
733         
734         for (File file : welcomeFiles) {
735 
736             String extension = FilenameUtils.getExtension(file.getCanonicalPath()).toLowerCase();
737 
738             Source source=new Source(file);
739             OutputDocument target=new OutputDocument(source);
740 
741             net.htmlparser.jericho.Element scriptElement =
742                     source.getElementById("isc-maven-plugin.reify-import.modifyWelcomeFiles");
743 
744             Set<String> projectNames = new LinkedHashSet<String>();
745             if (scriptElement != null) {
746                 if ("jsp".equals(extension)) {
747 
748                     String html = scriptElement.toString().trim();
749                     int start = html.toLowerCase().indexOf("name=\"") + 6;
750                     int end = html.indexOf("\"", start);
751                     String val = html.substring(start, end);
752 
753                     projectNames.addAll(Sets.newHashSet(val.split(",")));
754 
755                 } else {
756 
757                     String src = scriptElement.getAttributeValue("src");
758                     int start = src.indexOf("projectName=") + 12;
759                     int end = src.indexOf("&", start);
760                     String val = src.substring(start, end);
761 
762                     projectNames.addAll(Sets.newHashSet(val.split(",")));
763                 }
764             }
765             projectNames.add(projectName);
766 
767             String template = "LoadProjectScriptFragment.ftl";
768             if ("jsp".equals(extension)) {
769                 template = "LoadProjectTagFragment.ftl";
770             }
771 
772             Map context = ImmutableMap.of(
773                     "isomorphicDir", getIsomorphicDirPath(testHtmlPathname),
774                     "projects", projectNames.toArray(),
775                     "draw", drawOnWelcomeFiles);
776             StringWriter tagWriter = new StringWriter();
777             freemarkerConfig.getTemplate(template).process(context, tagWriter);
778 
779             if (scriptElement != null) {
780                 target.replace(scriptElement, tagWriter.toString());
781             } else {
782 
783                 // for now just drop the script at the end of the body
784                 int index = source.getFirstElement("body").getEndTag().getBegin();
785                 target.insert(index, tagWriter.toString());
786             }
787 
788             FileUtils.write(file, target.toString());
789         }
