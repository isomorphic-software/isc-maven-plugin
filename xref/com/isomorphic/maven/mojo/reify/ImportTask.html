14  /**
15   * An Ant task allowing the Reify {@link ImportMojo} to be run from Ant builds.  Note that
16   * default values have been changed to accomodate a typical Ant project structure, otherwise
17   * functionality is unchanged.
18   * <p>
19   * To use this Task, just add a taskdef to the desired build target, and make sure you have the
20   * required classpath entries set up correctly.  If you've built your project using one of the 
21   * Maven archetypes for either  
22   * <a href="https://www.smartclient.com/smartgwt-latest/javadoc/com/smartgwt/client/docs/MavenSupport.html">SmartGWT</a>
23   * or
24   * <a href="https://www.smartclient.com/smartclient-latest/isomorphic/system/reference/?id=group..mavenSupport">SmartClient</a>
25   * (and subsequently issued the ant 'unmaven' target), this should all be done for you.  
26   * Otherwise, you'll want to add something like the following to your build:
27   * <pre>
28   * &lt;property name="reify.lib" value="${basedir}/build/ivy/reify/lib" /&gt;
29   * 
30   * &lt;path id="reify.classpath"&gt;
31   *     &lt;fileset dir="${reify.lib}" erroronmissingdir="false"&gt;
32   *       &lt;include name="**&#47;*.jar" /&gt;
33   *     &lt;/fileset&gt;
34   * &lt;/path&gt;
35   * 
36   * &lt;target name="reify-tasklibs"&gt;
37   *     &lt;mkdir dir="${reify.lib}" /&gt;
38   *     &lt;ivy:resolve conf="reify" /&gt;
39   *     &lt;ivy:retrieve conf="reify" pattern="${reify.lib}/[artifact]-[revision](-[classifier]).[ext]"/&gt;
40   * &lt;/target&gt;
41   * 
42   * &lt;target name="reify-import" depends="reify-tasklibs"&gt;
43   * 
44   *     &lt;taskdef name="reify-import" classname="com.isomorphic.maven.mojo.reify.ImportTask" classpathref="reify.classpath"/&gt;
45   *     &lt;reify-import projectName="MyProject" username="${username}" password="${password}" 
46   *                   datasourcesDir="WEB-INF/ds/classic-models" 
47   *                   smartclientRuntimeDir="${basedir}/war/isomorphic" /&gt;
48   * &lt;/target&gt;
49   * </pre>
50   * Use of ivy is of course optional,  you just need somehow to get the isomorphic_m2pluginextras module and its 
51   * <a href="https://www.smartclient.com/smartclient-latest/isomorphic/system/reference/?id=group..javaModuleDependencies">dependencies</a>
52   * on your classpath - here they are copied to the directory defined in the <code>reify.lib</code> property)
53   * <p>
54   * Note that users working behind a proxy may need to provide server information to Ant 
55   * by way of the ANT_OPTS environment variable:
56   * <pre>
57   * export ANT_OPTS="-Dhttp.proxyHost=myproxyhost -Dhttp.proxyPort=8080 -Dhttp.proxyUser=myproxyusername -Dhttp.proxyPassword=myproxypassword"
58   * </pre>
59   */

97  		}
98

104 		mojo.setIncludeTestJsp(includeTestJsp);
105 		mojo.setTestJspPathname(testJspPathname = testJspPathname != null ? testJspPathname : projectName + ".run.jsp");
106 		mojo.setIncludeTestHtml(includeTestHtml);
107 		mojo.setTestHtmlPathname(testHtmlPathname = testHtmlPathname != null ? testHtmlPathname : projectName + ".run.html");
108 		mojo.setDataSourcesDir(dataSourcesDir);
109 		mojo.setMockDataSourcesDir(mockDataSourcesDir);
110 		mojo.setSkipValidationOnImport(skipValidationOnImport);
111 		mojo.setValidationFailureThreshold(validationFailureThreshold);
112 		mojo.setUiDir(uiDir);
113 		mojo.setProjectFileDir(projectFileDir);
114 		mojo.setModifyWelcomeFiles(modifyWelcomeFiles);
115 		mojo.setDrawOnWelcomeFiles(drawOnWelcomeFiles);
116 		mojo.setProjectName(projectName);
117 		mojo.setProjectFileName(projectFileName != null ? projectFileName : projectName + "proj.xml");
118 		mojo.setZipFileName(zipFileName != null ? zipFileName : projectName + "proj.zip");
119 
120 		// undocumented parameter to allow for testing against QA environment
121 		Object host = PropertyHelper.getProperty(getProject(), "host");
122 		if (host != null) {
123 			mojo.setHost(String.valueOf(host));
124 		}		
125 		
126 		String proxyHost = System.getProperty("http.proxyHost");
127 		String proxyPort = System.getProperty("http.proxyPort");
128 		String proxyUser = System.getProperty("http.proxyUser");
129 		String proxyPassword = System.getProperty("http.proxyPassword");
130 		
131 		//undocumented by Ant, allowed / expected by Maven so allow it here
132 		String proxyBypass = System.getProperty("http.nonProxyHosts");
133 
134 		if (proxyHost != null || proxyPort != null) {
135 			Proxy p = new Proxy();
136 			p.setHost(proxyHost != null ? proxyHost : "");
137 			p.setPort(Integer.valueOf(proxyPort != null ? proxyPort : "8080"));
138 			p.setUsername(proxyUser != null ? proxyUser : "");
139 			p.setPassword(proxyPassword != null ? proxyPassword : "");
140 			p.setNonProxyHosts(proxyBypass != null ? proxyBypass : "localhost");
141 			p.setActive(true);
142 			
143 			mojo.setProxy(p);			
144 		}
145 		
146 		try {
147 			mojo.execute();
148 		} catch (Exception e) {
149 			throw new BuildException(e);
150 		}
151 		
152 	}
