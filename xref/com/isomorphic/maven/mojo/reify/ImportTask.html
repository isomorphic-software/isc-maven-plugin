12  /**
13   * An Ant task allowing the Reify {@link ImportMojo} to be run from Ant builds.  Note that
14   * default values have been changed to accomodate a typical Ant project structure, otherwise
15   * functionality is unchanged.
16   * <p>
17   * To use this Task, just add a taskdef to the desired build target, and make sure you have the
18   * required classpath entries set up correctly.  If you've built your project using one of the 
19   * Maven archetypes for either  
20   * <a href="https://www.smartclient.com/smartgwt-latest/javadoc/com/smartgwt/client/docs/MavenSupport.html">SmartGWT</a>
21   * or
22   * <a href="https://www.smartclient.com/smartclient-latest/isomorphic/system/reference/?id=group..mavenSupport">SmartClient</a>
23   * (and subsequently issued the ant 'unmaven' target), this should all be done for you.  
24   * Otherwise, you'll want to add something like the following to your build:
25   * <pre>
26   * &lt;property name="reify.lib" value="${basedir}/build/ivy/reify/lib" /&gt;
27   * 
28   * &lt;path id="reify.classpath"&gt;
29   *     &lt;fileset dir="${reify.lib}" erroronmissingdir="false"&gt;
30   *       &lt;include name="**&#47;*.jar" /&gt;
31   *     &lt;/fileset&gt;
32   * &lt;/path&gt;
33   * 
34   * &lt;target name="reify-tasklibs"&gt;
35   *     &lt;mkdir dir="${reify.lib}" /&gt;
36   *     &lt;ivy:resolve conf="reify" /&gt;
37   *     &lt;ivy:retrieve conf="reify" pattern="${reify.lib}/[artifact]-[revision](-[classifier]).[ext]"/&gt;
38   * &lt;/target&gt;
39   * 
40   * &lt;target name="reify-import" depends="reify-tasklibs"&gt;
41   * 
42   *     &lt;taskdef name="reify-import" classname="com.isomorphic.maven.mojo.reify.ImportTask" classpathref="reify.classpath"/&gt;
43   *     &lt;reify-import projectName="MyProject" username="${username}" password="${password}" 
44   *                   datasourcesDir="WEB-INF/ds/classic-models" 
45   *                   smartclientRuntimeDir="${basedir}/war/isomorphic" /&gt;
46   * &lt;/target&gt;
47   * </pre>
48   * Use of ivy is of course optional,  you just need somehow to get the isomorphic_m2pluginextras module and its 
49   * <a href="https://www.smartclient.com/smartclient-latest/isomorphic/system/reference/?id=group..javaModuleDependencies">dependencies</a>
50   * on your classpath - here they are copied to the directory defined in the <code>reify.lib</code> property)
51   * <p>
52   * Note that users working behind a proxy may need to provide server information to Ant 
53   * by way of the ANT_OPTS environment variable:
54   * <pre>
55   * export ANT_OPTS="-Dhttp.proxyHost=myproxyhost -Dhttp.proxyPort=8080 -Dhttp.proxyUser=myproxyusername -Dhttp.proxyPassword=myproxypassword"
56   * </pre>
57   */
58  public class ImportTask extends Task {
59  
60      protected String serverUrl = "https://create.reify.com";
61      protected String workdir;
62      protected String webappDir = "war";
63      protected String smartclientRuntimeDir;
64      protected boolean includeTestJsp;
65      protected String testJspPathname;
66      protected boolean includeTestHtml;
67      protected String testHtmlPathname;
68      protected String dataSourcesDir = "WEB-INF/ds";
69      protected String mockDataSourcesDir = "WEB-INF/ds/mock";
70      protected boolean skipValidationOnImport;
71      protected Severity validationFailureThreshold = Severity.ERROR;
72      protected String uiDir = "WEB-INF/ui";
73      protected String projectFileDir = "WEB-INF/ui";
74      protected boolean modifyWelcomeFiles;
75      protected boolean drawOnWelcomeFiles;
76      protected String projectName;
77      protected String projectFileName;
78      protected String zipFileName;
79  
80      protected String username;
81      protected String password;
82  
83  
84      @Override
85      public void execute() throws BuildException {
86  
87          ImportMojo mojo = new ImportMojo();
88          mojo.setLog(new AntProjectLogger(getProject()));
89  
90          if (username == null || password == null) {
91              throw new BuildException("Username and password parameters required.");
92          }
93  
94          if (projectName == null) {
95              projectName = getProject().getName();
96          }
97  
98          mojo.setHost(String.valueOf(serverUrl));
99          mojo.setCredentials(new UsernamePasswordCredentials(username, password));
100 
101         mojo.setWorkdir(workdir != null ? new File(workdir) : new File(getProject().getBaseDir(), "build/reify"));
102         mojo.setWebappDir(webappDir != null ? new File(webappDir) : new File(getProject().getBaseDir(), webappDir));
103         mojo.setSmartclientRuntimeDir(smartclientRuntimeDir != null ? new File(smartclientRuntimeDir) : new File(getProject().getBaseDir(), "war/isomorphic"));
104         mojo.setIncludeTestJsp(includeTestJsp);
105         mojo.setTestJspPathname(testJspPathname = testJspPathname != null ? testJspPathname : projectName + ".run.jsp");
106         mojo.setIncludeTestHtml(includeTestHtml);
107         mojo.setTestHtmlPathname(testHtmlPathname = testHtmlPathname != null ? testHtmlPathname : projectName + ".run.html");
108         mojo.setDataSourcesDir(dataSourcesDir);
109         mojo.setMockDataSourcesDir(mockDataSourcesDir);
110         mojo.setSkipValidationOnImport(skipValidationOnImport);
111         mojo.setValidationFailureThreshold(validationFailureThreshold);
112         mojo.setUiDir(uiDir);
113         mojo.setProjectFileDir(projectFileDir);
114         mojo.setModifyWelcomeFiles(modifyWelcomeFiles);
115         mojo.setDrawOnWelcomeFiles(drawOnWelcomeFiles);
116         mojo.setProjectName(projectName);
117         mojo.setProjectFileName(projectFileName != null ? projectFileName : projectName + "proj.xml");
118         mojo.setZipFileName(zipFileName != null ? zipFileName : projectName + "proj.zip");
119 
120 
121         String proxyHost = System.getProperty("http.proxyHost");
122         String proxyPort = System.getProperty("http.proxyPort");
123         String proxyUser = System.getProperty("http.proxyUser");
124         String proxyPassword = System.getProperty("http.proxyPassword");
125 
126         //undocumented by Ant, allowed / expected by Maven so allow it here
127         String proxyBypass = System.getProperty("http.nonProxyHosts");
128 
129         if (proxyHost != null || proxyPort != null) {
130             Proxy p = new Proxy();
131             p.setHost(proxyHost != null ? proxyHost : "");
132             p.setPort(Integer.valueOf(proxyPort != null ? proxyPort : "8080"));
133             p.setUsername(proxyUser != null ? proxyUser : "");
134             p.setPassword(proxyPassword != null ? proxyPassword : "");
135             p.setNonProxyHosts(proxyBypass != null ? proxyBypass : "localhost");
136             p.setActive(true);
137 
138             mojo.setProxy(p);
139         }
140 
141         try {
142             mojo.execute();
143         } catch (Exception e) {
144             throw new BuildException(e);
145         }
146 
147     }
