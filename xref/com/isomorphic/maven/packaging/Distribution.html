144     /**
145      * Object creation method meant to facilitate a 'fluent interface' style of configuration via a handful of private methods.
146      * Objects are created with a set of default property values and retruned for further configuration if so deired.
147      * A reference to the return value is also available in the {@link #DISTRIBUTIONS} table under the given product/key combination.
148      *
149      * @param product
150      * @param license
151      * @param links
152      * @return A Distribution object having default configuration values applied
153      */
154     private static Distribution create(Product product, License license, String... links) {
155 
156         Distribution distribution = new Distribution(product, license);
157 
158         distribution.include(links);
159 
160         //e.g., smartclient downloads currently include smartgwt poms and vice-versa.  ignore when inapplicable
161         List<String> pomIncludes = new ArrayList<String>();
162         if (product == SMARTCLIENT) {
163             pomIncludes.add(POM_SMARTCLIENT);
164             distribution
165                     .contents("sdk/#smartclientSDK", "**/smartclientSDK/**", SMARTCLIENT_SDK_EXCLUDES)
166                     //exclude optional modules bundled with eval, and instead allow them to be repackaged in assemblies, as they would normally be (leave development copies of RTM in place for the dev console)
167                     .contents("assembly/smartclient-resources/#smartclientRuntime", SMARTCLIENT_RUNTIME_INCLUDES, "**/ISC_Analytics*,**/modules*/ISC_RealtimeMessaging*")
168                     .skins("/assembly/smartclient-resources.zip", "isomorphic/skins")
169                     //unfortunately, they're dropped on the root in optional download, modules directory in eval - use more specific source and target patterns to get the right file in the right place
170                     .contents("assembly/smartclient-analytics-resources/isomorphic/system/modules", "ISC_Analytics*,**/modules/ISC_Analytics*", null)
171                     .contents("assembly/smartclient-analytics-resources/isomorphic/system/modules-debug", "**/modules-debug/ISC_Analytics*", null)
172                     .contents("assembly/smartclient-messaging-resources/isomorphic/system/modules", "ISC_RealtimeMessaging*,**/modules/ISC_RealtimeMessaging*", null)
173                     .contents("assembly/smartclient-messaging-resources/isomorphic/system/modules-debug", "**/modules-debug/ISC_RealtimeMessaging*", null)
174                     .contents("assembly/smartclient-ai-resources/isomorphic/system/modules", "**/modules-debug/ISC_AI*", null)
175                     .contents("assembly/smartclient-ai-resources/isomorphic/system/modules-debug", "**/modules-debug/ISC_AI*", null)
176                     .contents("assembly/smartclient-tools-resources/#smartclientSDK", SMARTCLIENT_SDK_INCLUDES, SMARTCLIENT_SDK_EXCLUDES);
177         } else if (product == SMARTGWT) {
178 
179             //TODO check this against smartclient optional modules - how did it behave before?
180             distribution
181                 .contents("lib/smartgwt-analytics.jar", "**/analytics.jar", null)
182                 .contents("lib/smartgwt-messaging.jar", "**/messaging.jar", null)
183                 .contents("lib/smartgwt-ai.jar", "**/ai.jar", null)
184                 .skins("/lib/smartgwt-skins.jar", "com/smartclient/theme")
185                 .skins("/lib/smartgwt-" + license.getName() + ".jar", "com/smartclient/theme");
186 
187             pomIncludes.add(POM_SMARTGWT);
188         } else if (product == REIFY_ONSITE) {
189             // Handling skins in the REIFY_ONSITE goals is different from regular framework
190             // goals - see the custom impl in AbstractReifyOnSiteGoalsMojo
191             distribution.skins("REIFY_ONSITE", "isomorphic/skins");
192         }
193         pomIncludes.add("**/" + product.getName() + "-" + license.getName() + "*");
194         pomIncludes.add(POM_SHARED);
195 
196         //similarly lgpl includes server framework poms.  ignore
197         if (license != LGPL) {
198             pomIncludes.add(POM_SERVER);
199         }
200 
201         /*
202          * Map the relevant sdk resources to their paths as they should be when extracted.
203          * Note that some resources are renamed to provide clarity, facilitate POM matching,
204          * and/or adhere to conventions.
205          */
206         distribution
207                 .contents("pom", Joiner.on(",").join(pomIncludes), null)
208                 .contents("doc/user", DOC_INCLUDES, DOC_EXCLUDES)
209                 .contents("doc/api/client/#javadoc", SMARTGWT_CLIENT_JAVADOC, null)
210                 .contents("doc/api/server/#javadoc", product == SMARTGWT ? SMARTGWT_SERVER_JAVADOC : SMARTCLIENT_JAVADOC, null)
211                 .contents("lib", JAR_INCLUDES, JAR_EXCLUDES + ", " + JAR_CONFLICTS)
212                 .contents("lib/isc-jakarta-oro.jar", "**/isc-jakarta-oro*.jar", null)
213                 .contents("lib/isomorphic-messaging.jar", "**/isomorphic_realtime_messaging.jar", null)
214                 .contents("assembly/isc-selenium-resources", SELENIUM_INCLUDES, null);
215 
216         if (license == EVAL || license == POWER || license == ENTERPRISE) {
217             distribution.contents("assembly/isc-batchuploader-resources/ds", "**/batchUpload.ds.xml", null);
218         }
219 
220         DISTRIBUTIONS.put(product, license, distribution);
221         return distribution;
222     }

291     /**
292      * Adds an entry to the map of patterns used to determine which resources should be extracted from this distribution's collection of file/s.
293      *
294      * @param key A path to be used with any resource(s) that match the patterns provided.  Usually denotes a directory, but it can be used to specify a filename in cases where renaming is desirable.
295      * @param includes A comma-separated list of Ant-style patterns to be used to determine whether a given resource should be included in extraction / repackaging operations.
296      * @param excludes A comma-separated list of Ant-style patterns to be used to determine whether a given resource should be excluded from extraction / repackaging operations.
297      * @return This object, in 'fluid interface' style
298      *
299      * @see AntPathMatcherFilter
300      */
301     private Distribution contents(String key, String includes, String excludes) {
302         content.put(key, new AntPathMatcherFilter(includes, excludes));
303         return this;
304     }

306     /**
307      * Sets entries to the collection of regular expressions used to determine which hyperlinks should be used to download files from smartclient.com,
308      * clearing any expressions provided previously.
309      * <p>
310      * Each of the expressions provided is used to construct selection criteria used by an html parser.  Example:
311      * <pre>
312      *  	include("smartgwt-.*\\.zip");
313      * </pre>
314      * yields an expression like
315      * <pre>
316      * 		a[href~=smartgwt-.*\.zip]
317      * </pre>
318      * which will match links like the following:
319      * <pre>
320      * 		&lt;a href="/builds/SmartGWT/4.0p/LGPL/2014-01-08/smartgwt-4.0p.zip"&gt;smartgwt-4.0p.zip&lt;/a&gt;
321      * </pre>
322      *
323      * @param links The regular expressions used to determine which hyperlinks should be used to download files from smartclient.com
324      * @return This object, in 'fluid interface' style
325      * @see <a href="http://jsoup.org/cookbook/extracting-data/selector-syntax"></a>
326      */
327     private Distribution include(String... links) {
328         selectors.clear();
329         for (String pattern : links) {
330             selectors.add("a[href~=" + pattern + "]");
331         }
332         return this;
333     }

335     /**
336      * Performs token replacement on the URL template representing the location of the distribution's "remote index", or download page.
337      * Useful at initialization in the case where some product/license combination uses an unconventional URL.
338      *
339      * @param key the Key, including token, needing replacement.  e.g., #license
340      * @param replacement the value to use in the substitution.  e.g., EnterpriseEval
341      * @return This object, in 'fluid interface' style
342      */
343     private Distribution index(String key, String replacement) {
344         remoteIndex = remoteIndex.replaceAll(key, replacement);
345         return this;
346     }

348     /**
349      * Returns the relative URL representing the location of the distribution's "remote index", or download page.  e.g.,
350      * <p>
351      * /builds/SmartGWT/4.1d/EnterpriseEval/2014-01-01
352      *
353      * @param buildNumber  the buildNumber as found in the download url.  e.g., 4.1d
354      * @param buildDate the buildDate as found in the download url.  e.g., 2014-01-01
355      * @return the URL representing the location of the distribution's "remote index", or download page.
356      */
357     String getRemoteIndex(String buildNumber, String buildDate) {
358 
359         return remoteIndex
360                 .replaceAll("#product", product.toString())
361                 .replaceAll("#version", StringUtils.defaultString(buildNumber))
362                 .replaceAll("#license", license.toString())
363                 .replaceAll("#date", StringUtils.defaultString(buildDate));
364     }

366     /**
367      * Returns a comma-separated list of all the {@link #selectors} that should be used to determine which
368      *         hyperlinks should be used to download files from smartclient.com.
369      * @return a comma-separated list of all the {@link #selectors} that should be used to determine which
370      *         hyperlinks should be used to download files from smartclient.com
371      */
372     String getRemoteIndexFilter() {
373         return Joiner.on(",").join(selectors);

376     /**
377      * Extract the relevant contents from each file in the distribution.  Additionally creates ZIP/JAR
378      * files from specified resources (e.g., javadoc).
379      *
380      * @param to The directory to which each file should be extracted.
381      * @throws IOException on any IOException during unzip operations
382      */
383     public void unpack(File to) throws IOException {
384 
385         outer:
386         for (File file : files) {
387 
388             String ext = FilenameUtils.getExtension(file.getName()).toUpperCase();
389 
390             //copy uncompressed files to target, renaming as necessary per 'contents' configuration
391             if (! "ZIP".equals(ext)) {
392                 for(Map.Entry<String, AntPathMatcherFilter> filterEntry : content.entrySet()) {
393                     AntPathMatcherFilter filter = filterEntry.getValue();
394 
395                     if (filter.accept(file.getName())) {
396                         File target = FileUtils.getFile(to, ArchiveUtils.rewritePath(file.getName(), filterEntry.getKey()));
397 
398                         // filter jar for skins?
399 
400                         FileUtils.copyFile(file, target);
401                         LOGGER.debug("Copied file '{}' to file '{}'", file.getName(), target.getAbsolutePath());
402                         continue outer;
403                     }
404                 }
405                 FileUtils.copyFileToDirectory(file, new File(to, "lib"));
406                 continue outer;
407             }
408 
409             //otherwise extract contents (again renaming / relocating contents as necessary)
410             ZipFile zip = new ZipFile(file);
411             if (product != REIFY_ONSITE) {
412                 Enumeration<? extends ZipEntry> entries = zip.entries();
413 
414                 while (entries.hasMoreElements()) {
415                     ZipEntry entry = entries.nextElement();
416                     if (entry.isDirectory()) {  // OR entry matches skin exclusion
417                         continue;
418                     }
419                     for (Map.Entry<String, AntPathMatcherFilter> filterEntry : content.entrySet()) {
420                         AntPathMatcherFilter filter = filterEntry.getValue();
421                         if (filter.accept(entry.getName()) || product == REIFY_ONSITE) {
422                             File target = FileUtils.getFile(to, ArchiveUtils.rewritePath(entry.getName(), filterEntry.getKey()));
423                             FileUtils.copyInputStreamToFile(zip.getInputStream(entry), target);
424                             LOGGER.debug("Copied input stream to file '{}'", target.getAbsolutePath());
425                         }
426                     }
427                 }
428             } else {
429                 // The REIFY_ONSITE goals are completely different from regular framework install goals.
430                 // We don't need to be selective, or rename assets, or move them elsewhere - we can
431                 // just unzip
432                 Enumeration<? extends ZipEntry> entries = zip.entries();
433 
434                 while (entries.hasMoreElements()) {
435                     ZipEntry entry = entries.nextElement();
436                     if (entry.isDirectory()) {  // OR entry matches skin exclusion
437                         continue;
438                     }
439                     File target = FileUtils.getFile(to, entry.getName());
440                     FileUtils.copyInputStreamToFile(zip.getInputStream(entry), target);
441                     LOGGER.debug("Copied input stream to file '{}'", target.getAbsolutePath());
442                 }
443             }
444             zip.close();
445         }
446 
447         /*
448          * Create any number of assemblies by dropping their resources here.
449          * Each subdirectory will get zipped up and then deleted
450          */
451         File assembliesDir = new File(to, "assembly");
452 
453         List assemblies = CollectionUtils.arrayToList(assembliesDir.listFiles(new FileFilter() {
454             @Override
455             public boolean accept(File arg0) {
456                 return arg0.isDirectory();
457             }
458         }));
459         for (Object o : assemblies) {
460             File assembly = (File) o;
461             String name = FilenameUtils.getBaseName(assembly.getName());
462             LOGGER.debug("Copying resources for assembly '{}'", name);
463             ArchiveUtils.zip(assembly, FileUtils.getFile(assembliesDir, name + ".zip"));
464             FileUtils.deleteQuietly(assembly);
465         }
466 
467         LOGGER.debug("Repackaging Javadoc...");
468         File docLib = new File(to, "doc/lib");
469 
470         //TODO these paths should probably all be stuck in some constant
471         File client = FileUtils.getFile(to, "doc/api/client");
472         if (client.exists()) {
473             ArchiveUtils.jar(client, new File(docLib, "smartgwt-javadoc.jar"));
474         }
475 
476         File server = FileUtils.getFile(to, "doc/api/server");
477         if (server.exists()) {
478             ArchiveUtils.jar(server, new File(docLib, "isomorphic-javadoc.jar"));
479         }
480     }
