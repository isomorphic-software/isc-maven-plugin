47  /**
48   * Connects to Isomorphic site, discovers which files exist for a given build, and downloads them 
49   * to local file system. 
50   */
51  public class Downloads {
52  
53      private static final Logger LOGGER = LoggerFactory.getLogger(Downloads.class);
54      private static HttpRequestManager httpWorker;
55  
56      private File toFolder = new File(System.getProperty("java.io.tmpdir"));
57      private Boolean overwriteExistingFiles = Boolean.FALSE;
58  
59      /**
60       * Constructor taking the request manager used to communicate with smartclient.com.
61       *
62       * @param worker the request manager used to communicate with smartclient.com.
63       */
64      public Downloads(HttpRequestManager worker) {
65          httpWorker = worker;
66      }
67  
68      /**
69       * Sets the directory to which the distribution/s should be downloaded.
70       * Defaults to the system property <code>java.io.tmpdir</code>.
71       *
72       * @param toFolder The directory to which the distribution/s should be downloaded.
73       */
74      public void setToFolder(File toFolder) {
75          this.toFolder = toFolder;
76      }
77  
78      /**
79       * If true, downloads files whether they already exist locally or not.  Skips
80       * the download otherwise.  Defaults to false.
81       *
82       * @param overwriteExistingFiles true if files should be overwritten, false otherwise.
83       */
84      public void setOverwriteExistingFiles(Boolean overwriteExistingFiles) {
85          this.overwriteExistingFiles = overwriteExistingFiles;
86      }
87  
88      /**
89       * Retrieves a {@link Distribution} instance for each of the given licenses, downloads
90       * files if necessary, and {@link Distribution#getFiles() links} the local file to the distribution.
91       * <p>
92       * Refer to <a href="http://www.smartclient.com/builds/"></a>
93       *
94       * @param product The product built and distributed by Isomorphic Software.  e.g., SmartClient
95       * @param buildNumber The build number of the desired distribution.  e.g., 4.1d
96       * @param buildDate The date on which the distribution was made available
97       * @param licenses The licenses, or editions, that the product is released under, and for which the user is registered
98       * @return A collection of Distributions, each having its contents resolved to local files, and suitable for use in repacking operations
99       * @throws MojoExecutionException on any error
100      */
101     public List<Distribution> fetch(Product product, String buildNumber, String buildDate, License...licenses) throws MojoExecutionException {
102 
103         List<Distribution> result = new ArrayList<Distribution>();
104 
105         for (License license : licenses) {
106             Distribution distribution = Distribution.get(product, license);
107             download(distribution, buildNumber, buildDate);
108             result.add(distribution);
109         }
110 
111         return result;
112 
113     }
114 
115     public String findCurrentBuild(Distribution distribution, String buildNumber) throws MojoExecutionException {
116 
117         String url = distribution.getRemoteIndex(buildNumber, null);
118         String selector = "a[href~=[0-9]{4}-[0-9]{2}-[0-9]{2}]";
119         
120         String[] links = list(url, selector);
121 
122         if (links.length > 0) {
123             return links[0];
124         } else {
125             return null;
126         }
127     }
128 
129     /**
130      * Obtains a list of hyperlinks, downloads the file/s represented by each, and links it/them to the given distribution.
131      *
132      * @param distribution
133      * @throws MojoExecutionException
134      */
135     private void download (Distribution distribution, String buildNumber, String buildDate) throws MojoExecutionException {
136 
137         String url = distribution.getRemoteIndex(buildNumber, buildDate);
138         String selector = distribution.getRemoteIndexFilter();
139 
140         String[] links = list(url, selector);
141         
142         for (String link : links) {
143 
144             String filename = FilenameUtils.getName(link);
145 
146             if (distribution.product.equals(REIFY_ONSITE)) {
147                 if (filename.indexOf("-SmartClient") == -1) {
148                     // Skip the SmartGWT package in the same download location...
149                     continue;
150                 }
151             }
152 
153             File file = new File(toFolder, filename);
154 
155             if (file.exists() && !overwriteExistingFiles) {
156                 LOGGER.info("Existing archive found at '{}'.  Skipping download.", file.getAbsolutePath());
157                 distribution.getFiles().add(file);
158                 continue;
159             }
160 
161             HttpGet httpget = new HttpGet(link);
162             HttpResponse response;
163 
164             try {
165                 response = httpWorker.execute(httpget);
166             } catch (Exception e) {
167                 throw new MojoExecutionException("Error issuing GET request for bundle at '" + httpget + "'", e);
168             }
169 
170             HttpEntity entity = response.getEntity();
171 
172             if(!toFolder.mkdirs() && !toFolder.exists()) {
173                 throw new MojoExecutionException("Could not create specified working directory '" + toFolder.getAbsolutePath() + "'");
174             }
175 
176             FileUtils.deleteQuietly(file);
177 
178             OutputStream outputStream = null;
179 
180             try {
181                 LOGGER.info("Downloading file '{}' from '{}' to '{}'", filename, link, file.getAbsolutePath());
182                 outputStream = new LoggingCountingOutputStream(new FileOutputStream(file), entity.getContentLength());
183                 entity.writeTo(outputStream);
184                 distribution.getFiles().add(file);
185             } catch (Exception e) {
186                 throw new MojoExecutionException("Error writing file to '" + file.getAbsolutePath() + "'", e);
187             } finally {
188                 IOUtils.closeQuietly(outputStream);
189             }
190         }
191     }

193     /**
194      * Interrogates the remote server for a list of hyperlinks matching the given distribution's {@link Distribution#getRemoteIndexFilter() filter}.
195      *
196      * @return a String array of html href attributes
197      * @throws MojoExecutionException
198      */
199     private String[] list(String url, String selector) throws MojoExecutionException {
200 
201         HttpGet request = new HttpGet(url);
202         HttpResponse response;
203 
204         try {
205 
206             LOGGER.debug("Requesting list of files from '{}{}'", httpWorker.getHostName(), url);
207             response = httpWorker.execute(request);
208 
209         } catch (Exception e) {
210             throw new MojoExecutionException("Error issuing GET request for bundle at '" + request + "'", e);
211         }
212 
213         Document doc;
214 
215         try {
216 
217             String html = EntityUtils.toString(response.getEntity());
218             doc = Jsoup.parse(html);
219             doc.outputSettings().prettyPrint(true);
220 
221         } catch (Exception e) {
222             throw new MojoExecutionException("Error processing response from '" + request + "'", e);
223         }
224 
225         List<String> result = new ArrayList<String>();
226 
227         Elements links = doc.select(selector);
228 
229         for (Element element : links) {
230             String href = element.attr("href");
231             result.add(href);
232         }
233 
234         if (result.isEmpty()) {
235             String msg = String.format("No downloads found at '%s%s'.  Response from server: \n\n%s\n", httpWorker.getHostName(), url, doc.html());
236             LOGGER.warn(msg);
237         }
238 
239         return result.toArray(new String[0]);
240     }
