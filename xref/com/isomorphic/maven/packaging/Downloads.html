121 	/**
122 	 * Retrieves a {@link Distribution} instance for each of the given licenses, downloads
123 	 * files if necessary, and {@link Distribution#getFiles() links} the local file to the distribution.
124 	 * <p>
125 	 * Refer to http://www.smartclient.com/builds/
126 	 * 
127 	 * 
128 	 * @param product The product built and distributed by Isomorphic Software.  e.g., SmartCLient 
129 	 * @param buildNumber The build number of the desired distribution.  e.g., 4.1d
130 	 * @param buildDate The date on which the distribution was made available
131 	 * @param licenses The licenses, or editions, that the product is released under, and for which the user is registered
132 	 * @return A collection of Distributions, each having its contents resolved to local files, and suitable for use in repacking operations
133 	 * @throws MojoExecutionException on any error
134      */
135 	public List<Distribution> fetch(Product product, String buildNumber, String buildDate, License...licenses) throws MojoExecutionException {
136 
137 		try {
138 			setup();
139 			login();
140 			
141 			List<Distribution> result = new ArrayList<Distribution>();
142 			
143 			for (License license : licenses) {
144 				Distribution distribution = Distribution.get(product, license);
145 				download(distribution, buildNumber, buildDate);
146 				result.add(distribution);
147 			}
148 			logout();
149 
150 			return result;
151 			
152 		} finally {
153 	        httpClient.getConnectionManager().shutdown();
154 	    }
155 	}

157 	public String findCurrentBuild(Distribution distribution, String buildNumber) throws MojoExecutionException {
158 
159         try {
160             setup();
161             login();
162             
163             String url = distribution.getRemoteIndex(buildNumber, null);
164             String selector = "a[href~=[0-9]{4}-[0-9]{2}-[0-9]{2}]";
165             
166             String[] links = list(url, selector);
167             
168             logout();
169 
170             if (links.length > 0) {
171                 return links[0];
172             } else {
173                 return null;
174             }
175             
176         } finally {
177             httpClient.getConnectionManager().shutdown();
178         }

181 	/**
182 	 * Obtains a list of hyperlinks, downloads the file/s represented by each, and links it/them to the given distribution.
183 	 * 
184 	 * @param distribution
185 	 * @throws MojoExecutionException
186 	 */
187 	private void download (Distribution distribution, String buildNumber, String buildDate) throws MojoExecutionException {
188 
189 	    String url = distribution.getRemoteIndex(buildNumber, buildDate);
190         String selector = distribution.getRemoteIndexFilter();
191 		
192         String[] links = list(url, selector);
193         
194 		for (String link : links) {
195 		
196 			String filename = FilenameUtils.getName(link);
197 			File file = new File(toFolder, filename);
198 
199 			if (file.exists() && !overwriteExistingFiles) {
200 				LOGGER.info("Existing archive found at '{}'.  Skipping download.", file.getAbsolutePath());
201 				distribution.getFiles().add(file);
202 				continue;
203 			}
204 
205 			HttpGet httpget = new HttpGet(link);
206 			HttpResponse response;
207 			
208 			try {			
209 				response = httpClient.execute(host, httpget);
210 			} catch (Exception e) {
211 				throw new MojoExecutionException("Error issuing GET request for bundle at '" + httpget + "'", e);
212 			}
213 
214 			HttpEntity entity = response.getEntity();
215 			
216 			if(!toFolder.mkdirs() && !toFolder.exists()) {
217 				throw new MojoExecutionException("Could not create specified working directory '" + toFolder.getAbsolutePath() + "'");
218 	        }
219 			
220 			FileUtils.deleteQuietly(file);
221 
222             OutputStream outputStream = null;
223 
224             try {
225                 LOGGER.info("Downloading file to '{}'", file.getAbsolutePath());
226                 outputStream = new LoggingCountingOutputStream(new FileOutputStream(file), entity.getContentLength());
227                 entity.writeTo(outputStream);
228                 distribution.getFiles().add(file);
229             } catch (Exception e) {
230                 throw new MojoExecutionException("Error writing file to '" + file.getAbsolutePath() + "'", e);
231             } finally {
232                 IOUtils.closeQuietly(outputStream);
233             }
234 		}
235 	}

237 	/**
238 	 * Interrogates the remote server for a list of hyperlinks matching the given distribution's {@link Distribution#getRemoteIndexFilter() filter}.
239 	 * 
240 	 * @param dist the build in which some files should exist
241 	 * @return a String array of html href attributes
242 	 * @throws MojoExecutionException
243 	 */
244 	private String[] list(String url, String selector) throws MojoExecutionException {
245 		
246 		HttpGet request = new HttpGet(url);
247 		HttpResponse response;
248 		
249 		try {
250 		
251 			LOGGER.debug("Requesting list of files from {}{}", DOMAIN, url);
252 			response = httpClient.execute(host, request);
253 		
254 		} catch (Exception e) {
255 			throw new MojoExecutionException("Error issuing GET request for bundle at '" + request + "'", e);
256 		}
257 		
258 		Document doc;
259 
260 		try {
261 		
262 			String html = EntityUtils.toString(response.getEntity());
263 			doc = Jsoup.parse(html);
264 			doc.outputSettings().prettyPrint(true);
265 		
266 		} catch (Exception e) {
267 			throw new MojoExecutionException("Error processing response from '" + request + "'", e);
268 		}
269 
270 		List<String> result = new ArrayList<String>(); 
271 		
272 		Elements links = doc.select(selector);
273 
274 		for (Element element : links) {
275 			String href = element.attr("href");
276 			result.add(href);
277 		}
278 		
279 		if (result.isEmpty()) {
280 			String msg = String.format("No downloads found at '%s%s'.  Response from server: \n\n%s\n", DOMAIN, url, doc.html());
281 			LOGGER.warn(msg);
282 		}
283 		
284 		return result.toArray(new String[0]);
285 	}

287 	/**
288 	 * If {@link #credentials} have been supplied, uses them to autthenticate to the isomorphic web site,
289 	 * allowing download of protected resources.
290 	 * 
291 	 * @throws ClientProtocolException
292 	 * @throws IOException
293 	 */
294 	private void login() throws MojoExecutionException {
295 
296 		if (credentials == null) {
297 			return;
298 		}
299 		
300 		String username = credentials.getUserName();
301 		String password = credentials.getPassword();
302 		
303 		LOGGER.debug("Authenticating to '{}' with username: '{}'", DOMAIN + LOGIN_URL, username);
304 		
305         HttpPost login = new HttpPost(LOGIN_URL);
306 
307         List <NameValuePair> nvps = new ArrayList <NameValuePair>();
308         nvps.add(new BasicNameValuePair("USERNAME", username));
309         nvps.add(new BasicNameValuePair("PASSWORD", password));
310 
311     	try {
312 			login.setEntity(new UrlEncodedFormEntity(nvps));
313 			HttpResponse response = httpClient.execute(host, login);
314 			EntityUtils.consume(response.getEntity());
315 		} catch (IOException e) {
316 			throw new MojoExecutionException("Error during POST request for authentication", e);
317 		}
318 	}

337 	/**
338 	 * Configures the {@link #httpClient}, specifically with the current {@link #proxyConfiguration}. 
339 	 * 
340 	 * @throws MojoExecutionException
341 	 */
342 	private void setup() throws MojoExecutionException {
343 		try {
344 			if (proxyConfiguration != null && isProxied(proxyConfiguration) ) {
345 				if (proxyConfiguration.getUsername() != null) {
346 					httpClient.getCredentialsProvider().setCredentials(
347 						new AuthScope(proxyConfiguration.getHost(), proxyConfiguration.getPort()),
348 						new UsernamePasswordCredentials(proxyConfiguration.getUsername(), proxyConfiguration.getPassword()));	
349 				}
350 				HttpHost proxy = new HttpHost(proxyConfiguration.getHost(), proxyConfiguration.getPort());
351 				httpClient.getParams().setParameter(ConnRoutePNames.DEFAULT_PROXY, proxy);
352 			}
353 		} catch (Exception e) {
354 			throw new MojoExecutionException("Error obtaining Maven settings", e);
355 		}
356 	}

358 	/**
359 	 * Adapted from the Site plugin's AbstractDeployMojo to allow http operations through proxy.
360 	 * <p>
361 	 * Refer to
362 	 * http://maven.apache.org/guides/mini/guide-proxies.html
363 	 * <br>
364 	 * http://maven.apache.org/plugins/maven-site-plugin/xref/org/apache/maven/plugins/site/AbstractDeployMojo.html.
365 	 */
366 	private boolean isProxied(Proxy proxyConfig) throws MalformedURLException {
367 	    String nonProxyHostsAsString = proxyConfig.getNonProxyHosts();
368 
369 		for (String nonProxyHost : StringUtils.split(nonProxyHostsAsString, ",;|")) {
370 			
371 			if (StringUtils.contains(nonProxyHost, "*")) {
372 				
373 				// Handle wildcard at the end, beginning or middle of the nonProxyHost
374 				final int pos = nonProxyHost.indexOf('*');
375 				String nonProxyHostPrefix = nonProxyHost.substring(0, pos);
376 				String nonProxyHostSuffix = nonProxyHost.substring(pos + 1);
377 
378 				// prefix*
379 				if (StringUtils.isNotEmpty(nonProxyHostPrefix)
380 						&& DOMAIN.startsWith(nonProxyHostPrefix)
381 						&& StringUtils.isEmpty(nonProxyHostSuffix)) {
382 
383 					return false;
384 				}
385 				
386 				// *suffix
387 				if (StringUtils.isEmpty(nonProxyHostPrefix)
388 						&& StringUtils.isNotEmpty(nonProxyHostSuffix)
389 						&& DOMAIN.endsWith(nonProxyHostSuffix)) {
390 					
391 					return false;
392 				}
393 				
394 				// prefix*suffix
395 				if (StringUtils.isNotEmpty(nonProxyHostPrefix)
396 						&& DOMAIN.startsWith(nonProxyHostPrefix)
397 						&& StringUtils.isNotEmpty(nonProxyHostSuffix)
398 						&& DOMAIN.endsWith(nonProxyHostSuffix)) {
399 					
400 					return false;
401 				}
402 
403 			} else if (DOMAIN.equals(nonProxyHost)) {
404 				return false;
405 			}
406 		}
407 	    return true;
408 	}
