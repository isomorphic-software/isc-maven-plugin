46  /**
47   * Connects to Isomorphic site, discovers which files exist for a given build, and downloads them 
48   * to local file system.

87  	/**
88  	 * Retrieves a {@link Distribution} instance for each of the given licenses, downloads
89  	 * files if necessary, and {@link Distribution#getFiles() links} the local file to the distribution.
90  	 * <p>
91  	 * Refer to http://www.smartclient.com/builds/
92  	 * 
93  	 * 
94  	 * @param product The product built and distributed by Isomorphic Software.  e.g., SmartCLient 
95  	 * @param buildNumber The build number of the desired distribution.  e.g., 4.1d
96  	 * @param buildDate The date on which the distribution was made available
97  	 * @param licenses The licenses, or editions, that the product is released under, and for which the user is registered
98  	 * @return A collection of Distributions, each having its contents resolved to local files, and suitable for use in repacking operations
99  	 * @throws MojoExecutionException on any error
100      */
101 	public List<Distribution> fetch(Product product, String buildNumber, String buildDate, License...licenses) throws MojoExecutionException {
102 
103 		List<Distribution> result = new ArrayList<Distribution>();
104 		
105 		for (License license : licenses) {
106 			Distribution distribution = Distribution.get(product, license);
107 			download(distribution, buildNumber, buildDate);
108 			result.add(distribution);
109 		}
110 
111 		return result;
112 			
113 	}

115 	public String findCurrentBuild(Distribution distribution, String buildNumber) throws MojoExecutionException {
116 
117         String url = distribution.getRemoteIndex(buildNumber, null);
118         String selector = "a[href~=[0-9]{4}-[0-9]{2}-[0-9]{2}]";
119         
120         String[] links = list(url, selector);
121 
122         if (links.length > 0) {
123             return links[0];
124         } else {
125             return null;
126         }
127     }

129 	/**
130 	 * Obtains a list of hyperlinks, downloads the file/s represented by each, and links it/them to the given distribution.
131 	 * 
132 	 * @param distribution
133 	 * @throws MojoExecutionException
134 	 */
135 	private void download (Distribution distribution, String buildNumber, String buildDate) throws MojoExecutionException {
136 
137 	    String url = distribution.getRemoteIndex(buildNumber, buildDate);
138         String selector = distribution.getRemoteIndexFilter();
139 		
140         String[] links = list(url, selector);
141         
142 		for (String link : links) {
143 		
144 			String filename = FilenameUtils.getName(link);
145 			File file = new File(toFolder, filename);
146 
147 			if (file.exists() && !overwriteExistingFiles) {
148 				LOGGER.info("Existing archive found at '{}'.  Skipping download.", file.getAbsolutePath());
149 				distribution.getFiles().add(file);
150 				continue;
151 			}
152 
153 			HttpGet httpget = new HttpGet(link);
154 			HttpResponse response;
155 			
156 			try {			
157 				response = httpWorker.execute(httpget);
158 			} catch (Exception e) {
159 				throw new MojoExecutionException("Error issuing GET request for bundle at '" + httpget + "'", e);
160 			}
161 
162 			HttpEntity entity = response.getEntity();
163 			
164 			if(!toFolder.mkdirs() && !toFolder.exists()) {
165 				throw new MojoExecutionException("Could not create specified working directory '" + toFolder.getAbsolutePath() + "'");
166 	        }
167 			
168 			FileUtils.deleteQuietly(file);
169 
170             OutputStream outputStream = null;
171 
172             try {
173                 LOGGER.info("Downloading file to '{}'", file.getAbsolutePath());
174                 outputStream = new LoggingCountingOutputStream(new FileOutputStream(file), entity.getContentLength());
175                 entity.writeTo(outputStream);
176                 distribution.getFiles().add(file);
177             } catch (Exception e) {
178                 throw new MojoExecutionException("Error writing file to '" + file.getAbsolutePath() + "'", e);
179             } finally {
180                 IOUtils.closeQuietly(outputStream);
181             }
182 		}
183 	}

185 	/**
186 	 * Interrogates the remote server for a list of hyperlinks matching the given distribution's {@link Distribution#getRemoteIndexFilter() filter}.
187 	 * 
188 	 * @param dist the build in which some files should exist
189 	 * @return a String array of html href attributes
190 	 * @throws MojoExecutionException
191 	 */
192 	private String[] list(String url, String selector) throws MojoExecutionException {
193 		
194 		HttpGet request = new HttpGet(url);
195 		HttpResponse response;
196 		
197 		try {
198 		
199 			LOGGER.debug("Requesting list of files from '{}{}'", httpWorker.getHostName(), url);
200 			response = httpWorker.execute(request);
201 		
202 		} catch (Exception e) {
203 			throw new MojoExecutionException("Error issuing GET request for bundle at '" + request + "'", e);
204 		}
205 		
206 		Document doc;
207 
208 		try {
209 		
210 			String html = EntityUtils.toString(response.getEntity());
211 			doc = Jsoup.parse(html);
212 			doc.outputSettings().prettyPrint(true);
213 		
214 		} catch (Exception e) {
215 			throw new MojoExecutionException("Error processing response from '" + request + "'", e);
216 		}
217 
218 		List<String> result = new ArrayList<String>(); 
219 		
220 		Elements links = doc.select(selector);
221 
222 		for (Element element : links) {
223 			String href = element.attr("href");
224 			result.add(href);
225 		}
226 		
227 		if (result.isEmpty()) {
228 			String msg = String.format("No downloads found at '%s%s'.  Response from server: \n\n%s\n", httpWorker.getHostName(), url, doc.html());
229 			LOGGER.warn(msg);
230 		}
231 		
232 		return result.toArray(new String[0]);
233 	}
