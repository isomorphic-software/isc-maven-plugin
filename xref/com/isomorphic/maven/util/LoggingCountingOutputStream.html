1   package com.isomorphic.maven.util;
2   
3   import org.apache.commons.io.FileUtils;
4   import org.apache.commons.io.output.CountingOutputStream;
5   import org.apache.commons.lang3.StringUtils;
6   
7   import java.io.IOException;
8   import java.io.OutputStream;
9   
10  /**
11   * @author Daniel Johansson
12   * @since 2014-02-17 09:04
13   */
14  public class LoggingCountingOutputStream extends CountingOutputStream {
15  
16      private final long expectedByteCount;
17      private long snapshotStart = System.currentTimeMillis();
18      private long snapshotBytes;
19      private long bytesPerSecond;
20  
21      /**
22       * Constructs a new CountingOutputStream.
23       *
24       * @param out the OutputStream to write to
25       * @param expectedByteCount the number of bytes expected in the stream
26       */
27      public LoggingCountingOutputStream(final OutputStream out, final long expectedByteCount) {
28          super(out);
29          this.expectedByteCount = expectedByteCount;
30      }
31  
32      @Override
33      protected void afterWrite(final int bytesWritten) throws IOException {
34          super.afterWrite(bytesWritten);
35  
36          final long byteCount = getByteCount();
37          final double progress = ((double) byteCount / (double) expectedByteCount) * 100.0;
38  
39          snapshotBytes += bytesWritten;
40  
41          // Lets grab the number of bytes written over the last second and use that as our
42          // bytesPerSecond gauge giving the user an indication of speed.
43          if (System.currentTimeMillis() - snapshotStart >= 1000) {
44              snapshotStart = System.currentTimeMillis();
45              bytesPerSecond = snapshotBytes;
46              snapshotBytes = 0;
47          }
48  
49          // Using \r in this print out will cause the console to render this line of text on the
50          // same line in order to avoid console spam.
51          System.out.print("\r" + StringUtils.rightPad(FileUtils.byteCountToDisplaySize(byteCount) + " / " + FileUtils.byteCountToDisplaySize(expectedByteCount) + " (" + FileUtils.byteCountToDisplaySize(bytesPerSecond) + "/second) " + String.format("%.1f%%", progress), 60, " "));
52  
53          if (progress >= 100.0) {
54              System.out.println("Done!");
55          }
56      }
57  }
