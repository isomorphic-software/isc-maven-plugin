34  		output.getParentFile().mkdirs();

58  	public static String rewritePath(String source, String target) {
59  
60  		String path = target;
61  		String filename;
62  
63  		if ("".equals(FilenameUtils.getExtension(target))) {
64  			filename = FilenameUtils.getName(source);
65  		} else {
66  			path = FilenameUtils.getPath(target);
67  			filename = FilenameUtils.getName(target);
68  		}
69  		
70  		Pattern p = Pattern.compile("(.*?)#(.*?)(?:$|(/.*))");
71  		Matcher m = p.matcher(path);
72  		if (m.find()) {
73  			
74  			String prefix = m.group(1);
75  			String dir = m.group(2);
76  			String suffix = m.group(3);
77  
78  			int index = source.indexOf(dir);
79  			int length = dir.length();
80  			
81  			String remainder = FilenameUtils.getPath(source.substring(index + length));
82  			
83  			if (prefix != null && !prefix.equals("")) {
84  				path = prefix + remainder;				
85  			} else {
86  				path = remainder;
87  			}
88  			
89  			if (suffix != null && !suffix.equals("")) {
90  				path += suffix;				
91  			}
92  		}
93  		
94  		return FilenameUtils.normalize(path + "/" + filename);
95  	}

97  	private static void zip(File root, File source, ZipOutputStream target) throws IOException {
98  		String relativePath = root.toURI().relativize(source.toURI()).getPath().replace("\\", "/");
99  		
100 		BufferedInputStream in = null;
101 		try {
102 			if (source.isDirectory()) {
103 
104 				if (!relativePath.endsWith("/")) {
105 					relativePath += "/";
106 				}
107 				ZipEntry entry = ZipEntryFactory.get(target, relativePath);
108 				entry.setTime(source.lastModified());
109 				target.putNextEntry(entry);
110 				target.closeEntry();
111 
112 				for (File nestedFile : source.listFiles()) {
113 					zip(root, nestedFile, target);
114 				}
115 				return;
116 			}
117 
118 			ZipEntry entry = ZipEntryFactory.get(target, relativePath);
119 			entry.setTime(source.lastModified());
120 			target.putNextEntry(entry);
121 			in = new BufferedInputStream(FileUtils.openInputStream(source));
122 			IOUtils.copy(in, target);
123 			target.closeEntry();
124 
125 		} finally {
126 			IOUtils.closeQuietly(in);
127 		}
128 	}
