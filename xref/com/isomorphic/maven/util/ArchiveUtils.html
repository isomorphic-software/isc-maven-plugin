46  	/**
47  	 * Builds a JAR file from the contents of a directory on the filesystem (recursively).
48  	 * Adapted from stackoverflow solution.  
49  	 * <p>
50  	 * Refer to http://stackoverflow.com/questions/1281229/how-to-use-jaroutputstream-to-create-a-jar-file
51  	 * 
52  	 * @param directory the directory containing the content to be xzipped up
53  	 * @param output the zip file to be written to
54  	 */
55  	public static void jar(File directory, File output) throws IOException {
56  		Manifest manifest = new Manifest();
57  		manifest.getMainAttributes().put(Attributes.Name.MANIFEST_VERSION, "1.0");
58  		output.getParentFile().mkdirs();
59  		JarOutputStream target = new JarOutputStream(new FileOutputStream(output), manifest);
60  		zip(directory, directory, target);
61  		target.close();
62  	}

65  	 * Builds a ZIP file from the contents of a directory on the filesystem (recursively).
66  	 * <p>
67  	 * Refer to http://stackoverflow.com/questions/1281229/how-to-use-jaroutputstream-to-create-a-jar-file
68  	 * 
69  	 * @param directory the directory containing the content to be xzipped up
70  	 * @param output the zip file to be written to
71  	 * @throws IOException
72  	 */
73  	public static void zip(File directory, File output) throws IOException {
74  		output.getParentFile().mkdirs();
75  		ZipOutputStream target = new ZipOutputStream(new FileOutputStream(output));
76  		zip(directory, directory, target);
77  		target.close();

81  	 * Unzips the <code>source</code> file to the <code>target</code> directory.
82  	 * 
83  	 * @param source The file to be unzipped
84  	 * @param target The directory to which the source file should be unzipped
85  	 * @throws IOException
86  	 */
87  	public static void unzip(File source, File target) throws IOException {
88  		
89  		ZipFile zip = new ZipFile(source);
90  	    Enumeration<? extends ZipEntry> entries = zip.entries();
91  	    while (entries.hasMoreElements()) {
92  	        ZipEntry entry = entries.nextElement();
93  	        File file = new File(target,  entry.getName());
94  	        FileUtils.copyInputStreamToFile(zip.getInputStream(entry), file);
95  	    }
96  	}

98  	/**
99  	 * Derives a new file path, in whole or in part, from an existing path.  The following use cases are supported explicitly:
100 	 * <ul>	
101 	 * 	<li>
102 	 * 		Rename a file (example: smartgwt-lgpl.jar)
103 	 * 		<p>
104 	 * 		<code>
105 	 * 			ArchiveUtils.rewritePath("smartgwtee-4.1d/lib/smartgwt.jar", "smartgwt-lgpl.jar");	
106 	 * 		</code> 
107 	 * 	</li>
108 	 * 	<li>
109 	 * 		Move to another directory (example: target/smartgwt.jar)
110 	 *		<p>
111 	 * 		<code>
112 	 * 			ArchiveUtils.rewritePath("smartgwtee-4.1d/lib/smartgwt.jar", "target");
113 	 * 		</code>
114 	 * </li>
115 	 * <li>Move and rename (example: target/smartgwt-lgpl.jar)
116 	 * 		<p>
117 	 * 		<code>
118 	 * 			ArchiveUtils.rewritePath("smartgwtee-4.1d/lib/smartgwt.jar", "target/smartgwt-lgpl.jar"); 
119 	 * 		</code> 
120 	 * </li>
121 	 *  <li>Move to new root directory, preserving some part of the existing path
122 	 *  	<ul>
123 	 *  		<li>
124 	 *  			example: doc/api/com/isomorphic/servlet/IDACall.html
125 	 *  			<p>
126 	 *  			<code>
127 	 *  				ArchiveUtils.rewritePath("smartgwtee-4.1d/doc/javadoc/com/isomorphic/servlet/IDACall.html","doc/api/#javadoc");
128 	 *  			</code>
129 	 *    		</li>
130 	 *    		<li>
131 	 *  			example: doc/api/com/isomorphic/servlet/network/FileAssembly.html
132 	 *  			<p>
133 	 *  			<code>
134 	 * 	 				ArchiveUtils.rewritePath("smartgwtee-4.1d/doc/javadoc/com/isomorphic/servlet/CompressionFilter.html", "doc/api/#javadoc/network");
135 	 *  			</code>
136 	 *    		</li>
137 	 *  	</ul>
138 	 *   </li>
139 	 * </ul>
140 	 * 
141 	 * @param oldValue the existing path
142 	 * @param newValue the value to use for the new path, including optional tokens
143 	 * @return
144 	 */
145 	public static String rewritePath(String oldValue, String newValue) {
146 
147 		String path = newValue;
148 		String filename;
149 
150 		if ("".equals(FilenameUtils.getExtension(newValue))) {
151 			filename = FilenameUtils.getName(oldValue);
152 		} else {
153 			path = FilenameUtils.getPath(newValue);
154 			filename = FilenameUtils.getName(newValue);
155 		}
156 		
157 		Pattern p = Pattern.compile("(.*?)#(.*?)(?:$|(/.*))");
158 		Matcher m = p.matcher(path);
159 		
160 		if (m.find()) {
161 			
162 			String prefix = m.group(1);
163 			String dir = m.group(2);
164 			String suffix = m.group(3);
165 
166 			int index = oldValue.indexOf(dir);
167 			int length = dir.length();
168 			
169 			String remainder = FilenameUtils.getPath(oldValue.substring(index + length));
170 			
171 			if (prefix != null && !prefix.equals("")) {
172 				path = prefix + remainder;				
173 			} else {
174 				path = remainder;
175 			}
176 			
177 			if (suffix != null && !suffix.equals("")) {
178 				path += suffix;				
179 			}
180 		}
181 		
182 		return FilenameUtils.normalize(path + "/" + filename);
183 	}

185 	/**
186 	 * Steps common to archiving both zip and jar files, which include reading files from disk and using
187 	 * their contents to create {@link ZipEntry ZipEntries} and writing them to a ZipOutputStream.
188 	 * 
189 	 * @param root
190 	 * @param source
191 	 * @param target
192 	 * @throws IOException
193 	 */
194 	private static void zip(File root, File source, ZipOutputStream target) throws IOException {
195 		String relativePath = root.toURI().relativize(source.toURI()).getPath().replace("\\", "/");
196 		
197 		BufferedInputStream in = null;
198 		try {
199 			if (source.isDirectory()) {
200 
201 				if (!relativePath.endsWith("/")) {
202 					relativePath += "/";
203 				}
204 				ZipEntry entry = ZipEntryFactory.get(target, relativePath);
205 				entry.setTime(source.lastModified());
206 				target.putNextEntry(entry);
207 				target.closeEntry();
208 
209 				for (File nestedFile : source.listFiles()) {
210 					zip(root, nestedFile, target);
211 				}
212 				return;
213 			}
214 
215 			ZipEntry entry = ZipEntryFactory.get(target, relativePath);
216 			entry.setTime(source.lastModified());
217 			target.putNextEntry(entry);
218 			in = new BufferedInputStream(FileUtils.openInputStream(source));
219 			IOUtils.copy(in, target);
220 			target.closeEntry();
221 
222 		} finally {
223 			IOUtils.closeQuietly(in);
224 		}
225 	}
226 	
227 	private static class ZipEntryFactory {
228 		static ZipEntry get(ZipOutputStream target, String name) {
229 			if (target instanceof JarOutputStream) {
230 				return new JarEntry(name);
231 			}
232 			return new ZipEntry(name);
233 		}
